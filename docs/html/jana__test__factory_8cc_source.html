57 void jana_test_factory::Process(const std::shared_ptr<const JEvent>& aEvent)
58 {
59  // This factory will grab the JSourceObject and JSourceObject2 types created by
60  // the source. For each JSourceObject2 objects, it will create a jana_test
61  // object (product of this factory). For each of those, it submits a JTask which
62  // implements some busy work to represent calculations done to produce the jana_test
63  // objects.
64 
65  // Get the JSourceObject and JSourceObject2 objects
66  auto sobjs = aEvent->GetT<JSourceObject >();
67  auto sobj2s = aEvent->GetT<JSourceObject2>();
68 
69  // Create a jana_test object for each JSourceObject2 object
70  std::vector< std::shared_ptr<JTaskBase> > sTasks;
71  sTasks.reserve(sobj2s.size());
72  for( auto sobj2 : sobj2s )
73  {
74  auto jtest = new jana_test();
75  mData.push_back( jtest );
76 
77 #if 1
78  // Make a lambda that does busy work representing what would be done to
79  // to calculate the jana_test objects from the inout objects.
80  //
81  // n.b. The JTask mechanism is hardwired to pass only the JEvent reference
82  // as an argument. All other parameters must be passed as capture variables.
83  // (e.g. the jtest and sObj2 objects)
84  auto sMyLambda = [=](const std::shared_ptr<const JEvent> &aEvent) -> double
85  {
86  // Busy work
87  std::mt19937 sRandomGenerator; // need dedicated generator to avoid thread conflict
88  auto sTime = std::chrono::high_resolution_clock::now().time_since_epoch().count();
89  sRandomGenerator.seed(sTime);
90  double c = sobj2->GetHitE();
91  for(int i=0; i<1000; i++){
92  double a = sRandomGenerator();
93  double b = sqrt( a*pow(1.23, -a))/a;
94  c += b;
95  jtest->AddRandom(a); // add to jana_test object
96  }
97  jtest->SetE( c ); // set energy of jana_test object
98 
99  return c/10.0; // more complicated than one normally needs but demonstrates use of a return value
100  };
101 
102  // Make task shared_ptr (Return type of lambda is double)
103  auto sTask = std::make_shared< JTask<double> >();
104 
105  // Set the event and the lambda
106  sTask->SetEvent(aEvent);
107  sTask->SetTask(std::packaged_task< double(const std::shared_ptr<const JEvent>&) >(sMyLambda));
108 
109  // Move the task onto the vector (to avoid changing shptr ref count). casts at the same time
110  sTasks.push_back(std::move(sTask));
111 #endif
112  }
113 
114 #if 1
115  // Submit the tasks to the queues in the thread manager.
116  // This function won't return until all of the tasks are finished.
117  // This thread will help execute tasks (hopefully these) in the meantime.
118  (*aEvent).GetJApplication()->GetJThreadManager()->SubmitTasks(sTasks);
119 
120  // For purposes of example, we grab the return values from the tasks
121  double sSum = 0.0;
122  for( auto sTask : sTasks ){
123  auto &st = static_cast< JTask<double>& >(*sTask);
124  sSum += st.GetResult();
125  }
126 #endif
127 }
