53 //This class is used to generate generic factories for the given types.
54 //Generic: JFactory<DataType> factories with no tags or internal object construction.
55 //This is intended to be used by event sources, which just need the factories to hold data.
56 //The way this is done is by using recursion on the variadic template arguments.

114 template <typename HeadType, typename... TailTypes>
115 void JSourceFactoryGenerator<HeadType, TailTypes...>::GenerateFactories(JFactorySet *factory_set)
116 {
117  //Initialize vector
118  std::vector<JFactory*> sFactories;
119  sFactories.reserve(sizeof...(TailTypes) + 1);
120 
121  //Make factories and return
122  MakeFactories(std::back_inserter(sFactories));
123  for(auto fac : sFactories) factory_set->Add(fac);
124 }
125 
126 template <typename HeadType, typename... TailTypes>
127 template <typename ContainerType>
128 void JSourceFactoryGenerator<HeadType, TailTypes...>::MakeFactories(std::back_insert_iterator<ContainerType> aIterator)
129 {
130  //Create the factory for the head type
131  aIterator = static_cast<JFactory*>(new JFactoryT<HeadType>);
132  //Continue making factories for tail types (unless tail is empty)
133  JSourceFactoryGenerator<TailTypes...>::MakeFactories(aIterator);
134 }
