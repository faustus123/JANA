53 /// This is a base class for all event source generators. JANA implements
54 /// event sources in a modular way so that new types of sources can be
55 /// easily added. Typically, this just means different file formats, but
56 /// can also be networked or shared memory sources.
57 ///
58 /// One may subclass the JEventSource class directly, but it is recommended
59 /// to use the template class JEventSourceGeneratorT instead. That defines
60 /// defaults for the required methods based on the JEventSource class
61 /// the generator is for while still allowing to customization where
62 /// needed. See the JEventSourceGeneratorT documentation for details.
63 
64 class JEventSourceGenerator{
65  public:
66 
67  friend JEventSourceManager;
68 
69  JEventSourceGenerator(JApplication *app=nullptr):mApplication(app){}
70  virtual ~JEventSourceGenerator(){}
71 
72  // Default versions of these are defined in JEventSourceGeneratorT.h
73  virtual std::string GetType(void) const = 0; ///< Return name of the source type this will generate
74  virtual std::string GetDescription(void) const = 0; ///< Return description of the source type this will generate
75  virtual JEventSource* MakeJEventSource( std::string source ) = 0; ///< Create an instance of the source type this generates
76  virtual double CheckOpenable( std::string source ) = 0; ///< See JEventSourceGeneratorT for description
77 
78  protected:
79 
80  /// This is called by JEventSourceManager::AddJEventSourceGenerator which
81  /// itself is called by JApplication::Add(JEventSourceGenerator*). There
82  /// should be no need to call it from anywhere else.
83  void SetJApplication(JApplication *app){ mApplication = app; }
84 
85  JApplication* mApplication{nullptr};
86 };
