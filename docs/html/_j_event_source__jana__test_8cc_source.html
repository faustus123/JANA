74  /// Read an event (or possibly block of events) from the source return it.

76  // If an event was successfully read in, return kSUCCESS. If there
77  // are no more events in the source to read, return kNO_MORE_EVENTS.
78  // If the source has no events at the moment, but may later (e.g.
79  // a live stream) then return kTRY_AGAIN;
80 
81  if( mNumEventsToGenerate != 0 )

84  //These are recycled, so be sure to re-set EVERY member variable
85  auto sEvent = mEventPool.Get_SharedResource(mApplication);
86  mNumEventsGenerated++;
87 
88  if( mIncludeBarriers ){
89  auto sIsBarrierEvent = (mNumEventsGenerated % 100) == 0;
90  sEvent->SetIsBarrierEvent(sIsBarrierEvent);
91  }
92  sEvent->SetEventNumber(mNumEventsGenerated);
93  sEvent->SetRunNumber(1234);
94

124 bool JEventSource_jana_test::GetObjects(const std::shared_ptr<const JEvent>& aEvent, JFactory* aFactory)
125 {
126  //Get objects of the specified type, and set them in the factory
127  //If this type is not supplied by the file, return false. Else return true.
128 
129  //For all of the objects retrievable from the file, we MUST generate factories via our own JFactoryGenerator
130  auto sTypeIndex = aFactory->GetObjectType();
131  if(sTypeIndex == std::type_index(typeid(JSourceObject)))
132  return GetObjects(aEvent, static_cast<JFactoryT<JSourceObject>*>(aFactory));
133  else if(sTypeIndex == std::type_index(typeid(JSourceObject2)))
134  return GetObjects(aEvent, static_cast<JFactoryT<JSourceObject2>*>(aFactory));
135  else
136  return false;
137 }

142 bool JEventSource_jana_test::GetObjects(const std::shared_ptr<const JEvent>& aEvent, JFactoryT<JSourceObject>* aFactory)
143 {
144  if(aFactory->GetTag() != "") return false; //Only default tag here
145 
146  // Generate a random # of objects
147  auto sNumObjectsDistribution = std::uniform_int_distribution<std::size_t>(1, 20);
148  auto sNumObjects = sNumObjectsDistribution(gRandomGenerator);
149  std::vector<JSourceObject*> sObjects;
150  for(std::size_t si = 0; si < sNumObjects; si++)
151  {
152  // Create new JSourceObject
153  auto sObject = new JSourceObject(gRandomGenerator(), si);
154  sObjects.push_back( sObject );
155 
156  // Supply busy work to take time. This would represent
157  // the parsing of the input data.
158  auto sNumRandomsDistribution = std::uniform_int_distribution<std::size_t>(1000, 2000);
159  auto sNumRandoms = sNumRandomsDistribution(gRandomGenerator);
160  for(std::size_t sj = 0; sj < sNumRandoms; sj++) sObject->AddRandom(gRandomGenerator());
161  }
162 
163  //Set the objects in the factory
164  aFactory->Set(std::move(sObjects));
165  return true;
166 }
