11 // This method gets called to actually produce the MyCluster objects for the event.
12 // We use "partial template specialization" to define this here in place of the default
13 // one defined in the templated JFactoryT class.
14 template<>
15 void JFactoryT<MyCluster>::Process(const std::shared_ptr<const JEvent>& aEvent)
16 {
17  // Get the MyHit objects for the event. We can grab as many different types of
18  // objects we need to here in order to make the MyCluster objects. For this
19  // example, only the MyHit objects are needed. Note how we do not need to specify
20  // whether these came from a JEventSource or another JFactory (i.e. algorithm).
21  // The type of the hits variable is a vector<const MyHit*> here. Using the
22  // "auto" type means we only need to type "MyHit" once, minimizing potential
23  // type conflicts from copy/paste errors.
24  auto hits = aEvent->GetT<MyHit>();
25 
26  // For simplicity, split hits into 3 groups and we'll make a MyCluster from each
27  vector< decltype(hits) > sorted_hits(3); // i.e. vector< vector<const MyHit*> > with 3 elements
28  for(size_t i=0; i<hits.size(); i++){
29  sorted_hits[i%3].push_back( hits[i] ); // copy pointer to MyHit to appropriate internal vector
30  }
31 
32  // Loop over groups of hits and make a MyCluster of each
33  for( auto vhits : sorted_hits ){ // loop over groups of hits (should be 3)
34 
35  if( vhits.empty() ) continue; // ignore empty groups
36 
37  // Fancy way of adding energy of all hits and averaging x and t
38  double Etot = std::accumulate(vhits.begin(), vhits.end(), 0.0, [](double s, const MyHit *hit){return s+hit->E;} );
39  double x_avg = std::accumulate(vhits.begin(), vhits.end(), 0.0, [](double s, const MyHit *hit){return s+hit->x;} )/(double)vhits.size();
40  double t_avg = std::accumulate(vhits.begin(), vhits.end(), 0.0, [](double s, const MyHit *hit){return s+hit->t;} )/(double)vhits.size();
41 
42  // Create a new MyCluster object
43  auto cluster = new MyCluster(x_avg, Etot, t_avg);
44 
45  // Objects of the type this factory creates (MyCluster in this case) should be
46  // stored in the mData member of the JFactoryT class. Note that this transfers
47  // ownership to the framework and JANA will take care of deleting these later.
48  mData.push_back( cluster );
49 
50  // One may add JObjects associated with this one to its associated objects list.
51  // This is a convenient way to keep the list of pointers to the MyHit objects
52  // used to create this cluster. You may also choose to do this in a more visible
53  // member variable if you prefer. Adding them as "associated objects" though makes
54  // JANA aware of the association, unlocking other JANA features that make use of
55  // that.
56  for( auto hit : vhits ) cluster->AddAssociatedObject( hit );
57  }
58 }
