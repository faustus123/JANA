136 // CreateSources
137 //---------------------------------
138 void JEventSourceManager::CreateSources(void)
139 {
140  //Single-threaded here!
141 
142  //Create all of the JEventSource's that we will need, even if they won't all be active at the same time.
143  //We will have a separate method to open the sources that we'll have to call before reading from them.
144 
145  //Why do we do this?
146  //We need to store the objects in the event sources in factories, and these types are only known to the source.
147  //These factories need to be created for these sources, and stored in the JFactorySet
148  //However, we want this operation to happen at the beginning of the program, while we're still single-threaded.
149  //If we instead do it on-demand during JEvent analysis, we'd have to lock the JFactorySet every time we add/get a factory from it.
150 
151  for(auto sSourceName : _source_names)
152  {
153  auto sSource = CreateSource(sSourceName);
154  if(sSource != nullptr)
155  _sources_unopened.push_back(sSource);
156  }
157 
158  if(_sources_unopened.empty())
159  {
160  jerr<<" xxxxxxxxxxxx NO VALID EVENT SOURCES GIVEN !!! xxxxxxxxxxxx "<<std::endl;
161  jerr<<std::endl;
162  mApplication->SetExitCode(-1);
163  mApplication->Quit();
164  }
165 }

168 // OpenInitSources
169 //---------------------------------
170 void JEventSourceManager::OpenInitSources(void)
171 {
172  //Single-threaded here!
173  if(_sources_unopened.empty())
174  return;
175 
176  //Open up to the max allowed number of simultaneously-open sources, and register them as active.
177  auto sNumFilesToOpen = (_sources_unopened.size() >= mMaxNumOpenFiles) ? mMaxNumOpenFiles : _sources_unopened.size();
178  for(std::size_t si = 0; si < sNumFilesToOpen; si++)
179  {
180  auto sSource = _sources_unopened.front();
181  _sources_unopened.pop_front();
182  sSource->Open();
183  _sources_active.push_back(sSource);
184  }
185 }

188 // CreateSource
189 //---------------------------------
190 JEventSource* JEventSourceManager::CreateSource(const std::string& source_name)
191 {
192  // Check if the user has forced a specific type of event source
193  // be used via the EVENT_SOURCE_TYPE config. parameter. If so,
194  // search for that source and use it. Otherwise, throw an exception.
195  auto gen = GetUserEventSourceGenerator();
196 
197  if(gen == nullptr)
198  gen = GetEventSourceGenerator(source_name);
199 
200  // Try opening the source using the chosen generator
201  JEventSource *new_source = nullptr;
202  if(gen != nullptr){
203  jout << "Opening source \"" << source_name << "\" - "<< gen->GetType() << " : " << gen->GetDescription() << endl;
204  new_source = gen->MakeJEventSource(source_name);
205  }
206 
207  if(new_source != nullptr){
208  new_source->SetJApplication(mApplication);
209  _sources_allocated.push_back(std::shared_ptr<JEventSource>(new_source)); // ensure destruction
210  return new_source;
211  }
212 
213  // Problem opening source. Notify user
214  jerr<<std::endl;
215  jerr<<" xxxxxxxxxxxx Unable to open event source \""<<source_name<<"\"! xxxxxxxxxxxx"<<std::endl;
216  jerr<<std::endl;
217 
218  return nullptr;
219 }

222 // OpenNext
223 //---------------------------------
224 std::pair<JEventSource::RETURN_STATUS, JEventSource*> JEventSourceManager::OpenNext(const JEventSource* aPreviousSource)
225 {
226  /// Remove the previous source from the opened list and open a new one.
227  auto sFindSlot = [aPreviousSource](const JEventSource* sSource) -> bool { return (sSource == aPreviousSource); };
228 
229  // Lock
230  std::lock_guard<std::mutex> lg(mSourcesMutex);
231 
232  //Check if a previous thread has already closed the last source
233  if(_sources_active.empty())
234  return std::make_pair(JEventSource::RETURN_STATUS::kNO_MORE_EVENTS, (JEventSource*)nullptr);
235 
236  //Find slot
237  auto sEnd = std::end(_sources_active);
238  auto sIterator = std::find_if(std::begin(_sources_active), sEnd, sFindSlot);
239  if(sIterator == sEnd) //Previous source already closed and new one already opened (if any)
240  return std::make_pair(JEventSource::RETURN_STATUS::kUNKNOWN, (JEventSource*)nullptr); //unknown: don't know if any were opened
241 
242  //Register source finished
243  _sources_exhausted.push_back(*sIterator);
244  _sources_active.erase(sIterator);
245 
246  //If no new sources to open, return
247  if(_sources_unopened.empty())
248  return std::make_pair(JEventSource::RETURN_STATUS::kNO_MORE_EVENTS, (JEventSource*)nullptr);
249 
250  //Get the new source
251  auto sNewSource = _sources_unopened.front();
252  _sources_unopened.pop_front();
253 
254  //Open the new source, register it, and return it
255  sNewSource->Open();
256  _sources_active.push_back(sNewSource);
257  return std::make_pair(JEventSource::RETURN_STATUS::kSUCCESS, sNewSource);
258 }

278 // GetEventSourceGenerator
279 //---------------------------------
280 JEventSourceGenerator* JEventSourceManager::GetEventSourceGenerator(const std::string& source_name)
281 {
282  // Loop over JEventSourceGenerator objects and find the one
283  // (if any) that has the highest chance of being able to read
284  // this source. The return value of
285  // JEventSourceGenerator::CheckOpenable(source) is a liklihood that
286  // the named source can be read by the JEventSource objects
287  // created by the generator. In most cases, the liklihood will
288  // be either 0.0 or 1.0. In the case that 2 or more generators return
289  // equal liklihoods, the first one in the list will be used.
290  JEventSourceGenerator* gen = nullptr;
291  double liklihood = 0.0;
292  for( auto sg : _eventSourceGenerators ){
293  double my_liklihood = sg->CheckOpenable(source_name);
294  if(my_liklihood > liklihood){
295  liklihood = my_liklihood;
296  gen = sg;
297  }
298  }
299  return gen;
300 }

304 //---------------------------------
305 JEventSourceGenerator* JEventSourceManager::GetUserEventSourceGenerator(void)
306 {
307  // Check if the user has forced a specific type of event source
308  // be used via the EVENT_SOURCE_TYPE config. parameter. If so,
309  // search for that source and use it. Otherwise, throw an exception.
310 
311  JEventSourceGenerator* gen = nullptr;
312  try{
313  std::string EVENT_SOURCE_TYPE = mApplication->GetParameterValue<string>("EVENT_SOURCE_TYPE");
314  for( auto sg : _eventSourceGenerators ){
315  if( sg->GetType() == EVENT_SOURCE_TYPE ){
316  gen = sg;
317  jout << "Forcing use of event source type: " << EVENT_SOURCE_TYPE << endl;
318  break;
319  }
320  }
321  if(!gen){
322  jerr << endl;
323  jerr << "-----------------------------------------------------------------" << endl;
324  jerr << " You specified event source type \"" << EVENT_SOURCE_TYPE << "\"" << endl;
325  jerr << " be used to read the event sources but no such type exists." << endl;
326  jerr << " Here is a list of available source types:" << endl;
327  jerr << endl;
328  for( auto sg : _eventSourceGenerators ) jerr << " " << sg->GetType() << endl;
329  jerr << endl;
330  jerr << "-----------------------------------------------------------------" << endl;
331  mApplication->SetExitCode(-1);
332  mApplication->Quit();
333  }
334  }catch(...){}
335 
336  return gen;
337 }

341 //---------------------------------
342 std::size_t JEventSourceManager::GetNumEventsProcessed(void) const
343 {
344  // Lock
345  std::lock_guard<std::mutex> lg(mSourcesMutex);
346 
347  auto sNumEventGetter = [](std::size_t aTotal, JEventSource* aSource) -> std::size_t {return aTotal + aSource->GetNumEventsProcessed();};
348 
349  //Sum from active sources and exhausted ones
350  auto sNumEvents = std::accumulate(std::begin(_sources_active), std::end(_sources_active), 0, sNumEventGetter);
351  return std::accumulate(std::begin(_sources_exhausted), std::end(_sources_exhausted), sNumEvents, sNumEventGetter);
352 }
