132 // JQueueWithBarriers
133 //---------------------------------
134 void JQueueWithBarriers::ThreadLoop(void)
135 {
136  //This thread is responsible for moving tasks from the input queue to the output queue.
137  //It will do so until it encounters a barrier event.
138  //At that point, no further tasks are moved into the output queue until the task with the barrier event is done.
139 
140  //This way, future events from this event source have the complete set of data from the barrier event.
141  //Otherwise, another thread could analyze the next event and not have the complete set of information.
142  //Note that every task in this queue is from the same event source.
143 
144  //We also hold onto a pointer to this barrier event, so that we can set it as a member of future events.
145  //That way we don't have to wait for events to flush out of the system before analyzing the barrier event:
146  //We are setting the data right here, where only this thread has access.
147 
148  //Unforunately, we can't use a single JQueue to do this.
149  //This is because, when getting tasks, once we get exclusive access to a slot, another thread
150  //is able to get the task from the following slot.
151  //We would need to check whether we had a barrier event and prevent the subsequent GetTask in a single operation
152  //which is impossible. Otherwise, we'd have to lock a mutex in JQueue.
153  //Thus we have this thread, which moves tasks between two queues.
154 
155  while(!mEndThread)
156  {
157  //Move all tasks from the input queue to the output queue, until we hit a barrier
158  while(!mEndThread)
159  {
160  //Get task from input queue
161  auto sTask = mInputQueue->GetTask();
162  if(sTask == nullptr)
163  break; //No more tasks to move
164 
165  //Get a non-const version of the event.
166  //Why non-const? So we can set the latest barrier event
167  auto sEvent = const_cast<JEvent*>(sTask->GetEvent());
168 
169  //Is it a barrier event?
170  auto sIsBarrierEvent = sEvent->GetIsBarrierEvent();
171 // std::cout << "sIsBarrierEvent = " << sIsBarrierEvent << std::endl;
172  if(sIsBarrierEvent)
173  {
174  //Hold a weak pointer to the task that is responsible for running the plugins over this event
175  mAnalyzeBarrierEventTask = sTask;
176 
177  //Also, keep hold of the event so we can add it to subsequent events
178  mLatestBarrierEvent = sTask->GetSharedEvent();
179 
180  //For the barrier event itself, set the latest barrier event to nullptr
181  //(if we store shared_ptr to self, it will never go out of scope)
182  sEvent->SetLatestBarrierEvent(nullptr);
183  }
184  else //Set the latest barrier event
185  sEvent->SetLatestBarrierEvent(mLatestBarrierEvent);
186 
187  //Move task to output queue
188  while( mOutputQueue->AddTask( std::move(sTask) ) == JQueue::Flags_t::kQUEUE_FULL ){
189  // Crap, the queue is full. Try pulling a task off and running it here.
190  // Then we can try adding our task again.
191  auto sEventTask = mOutputQueue->GetTask();
192  if( sEventTask != nullptr ){
193  (*sEventTask)();
194  mOutputQueue->AddTasksProcessedOutsideQueue(1);
195  }
196  }
197 
198  //If barrier, Don't move any more tasks to the output queue until the barrier event is finished being analyzed.
199  if(sIsBarrierEvent)
200  {
201  //When is it done being analyzed? When the weak_ptr has expired
202  while(!mAnalyzeBarrierEventTask.expired() && !mEndThread)
203  std::this_thread::sleep_for(mSleepTimeIfBarrier); //Wait
204  mAnalyzeBarrierEventTask.reset();
205 // std::cout << "COUNT IS ZERO!!!" << std::endl;
206  }
207  }
208 
209  //All tasks moved, sleep a while to wait for more
210  std::this_thread::sleep_for(mSleepTime);
211  }
212 }
