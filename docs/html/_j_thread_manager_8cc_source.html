110 void JThreadManager::CreateThreads(std::size_t aNumThreads)
111 {
112  //Create the threads, assigning them queue sets (and event sources)
113  //Before calling this, the queues (& event sources) should be prepared first!
114  mThreads.reserve(aNumThreads);
115  for(decltype(aNumThreads) si = 0; si < aNumThreads; si++)
116  {
117  auto sQueueSetIndex = si % mActiveSourceInfos.size();
118  mThreads.push_back(new JThread(si + 1, mApplication, mActiveSourceInfos[sQueueSetIndex], sQueueSetIndex, mRotateEventSources));
119  }
120 
121  // Tell JApplication to adjust maximum size of JFactorySet resource pool
122  mApplication->UpdateResourceLimits();
123 }

128 void JThreadManager::AddThread(void)
129 {
130  //Find a non-null event source info
131  std::size_t sSourceIndex = 0;
132  auto sEventSourceInfo = GetNextSourceQueues(sSourceIndex);
133 
134  if(sEventSourceInfo == nullptr)
135  return; //Weird. We must be ending the program. Or starting it. Don't add threads this way in these situations.
136 
137  //LOCK
138  LockThreadPool();
139 
140  mThreads.push_back(new JThread(mThreads.size() + 1, mApplication, sEventSourceInfo, sSourceIndex, mRotateEventSources));
141 
142  //UNLOCK
143  mThreadPoolLock = false;
144 
145  // Tell JApplication to adjust maximum size of JFactorySet resource pool
146  mApplication->UpdateResourceLimits();
147 }

174 // SetNJThreads
175 //---------------------------------
176 void JThreadManager::SetNJThreads(std::size_t nthreads)
177 {
178  // Force reasonable limit to number of threads
179  if( nthreads<1 || nthreads>4096 )throw JException("nthreads (%d) out of range! must be between 1-4096", nthreads);
180 
181  // Increase to desired number of threads
182  std::size_t sSourceIndex = 0;
183  while( GetNJThreads() < nthreads ) {
184  if( GetNextSourceQueues(sSourceIndex) == nullptr ) return; // Thread won't be added if this is null
185  AddThread();
186  }
187 
188  // Decrease to desired number of threads
189  while( GetNJThreads() > nthreads ) {
190  RemoveThread();
191  }
192 
193  // Tell JApplication to adjust maximum size of JFactorySet resource pool
194  mApplication->UpdateResourceLimits();
195 
196  // Make sure all threads are running
197  RunThreads();
198 }

204 {

214  mActiveSourceInfos.push_back(new JEventSourceInfo(sSource, MakeQueueSet(sSource)));

217  mRotateEventSources = false; //nothing to rotate to

220 //---------------------------------
221 // MakeQueueSet
222 //---------------------------------
223 JQueueSet* JThreadManager::MakeQueueSet(JEventSource* sEventSource)
224 {
225  //Create queue set from template
226  //The template contains queues that are identical for all sources (e.g. subtask, output queues)
227 
228  //Don't just use default copy-constructor.
229  //We want each event source to have its own set of queues.
230  //This is in case there are "barrier" events:
231  //If a barrier event comes through, we don't want to analyze any
232  //subsequent events until it is finished being analyzed.
233  //However, we can still analyze events from other sources,
234  //and don't want to block those: Separate queues
235  auto sQueueSet = mTemplateQueueSet.Clone();
236  mAllocatedQueueSets.push_back( std::shared_ptr<JQueueSet>(sQueueSet)); // ensure destructor is called later
237 
238  //get source-specific event queue (e.g. disentangle events)
239  auto sEventQueue = sEventSource->GetEventQueue();
240  if(sEventQueue == nullptr){
241  //unspecified by source, use default
242  sEventQueue = new JQueueSimple(mApplication->GetJParameterManager(),
243  "Events");
244  }
245 
246  // This will add the queue used by the JEventSource to output completed
247  // (parsed) events. The source may use other queues internally ahead of
248  // this one. n.b. JANA will place events obtained from the source via
249  // GetEvent into the first "Events" type queue in the set.
250  sQueueSet->AddQueue(JQueueSet::JQueueType::Events, sEventQueue);
251 
252  return sQueueSet;
253 }

360 //---------------------------------
361 JThreadManager::JEventSourceInfo* JThreadManager::GetEventSourceInfo(const JEventSource* aEventSource) const
362 {
363  auto sFind_Key = [aEventSource](const JEventSourceInfo* aSourceInfo) -> bool
364  {
365  if(aSourceInfo == nullptr)
366  return false;
367  return (aSourceInfo->mEventSource == aEventSource);
368  };
369 
370  //LOCK
371  LockScourceInfos();
372 
373  auto sEnd = std::end(mActiveSourceInfos);
374  auto sIterator = std::find_if(std::begin(mActiveSourceInfos), sEnd, sFind_Key);
375  auto sResult = (sIterator != sEnd) ? *sIterator : nullptr;
376 
377  //UNLOCK
378  mScourceInfosLock = false;
379 
380  return sResult;
381 }

383 //---------------------------------
384 // GetNextQueueSet
385 //---------------------------------
386 JThreadManager::JEventSourceInfo* JThreadManager::GetNextSourceQueues(std::size_t& aInputSetIndex)
387 {
388  //Rotate to the next event source (and corresponding queue set)
389  //Increment the current set index, and if past the end, loop back to 0
390  auto aCurrentSetIndex = aInputSetIndex + 1;
391 
392  //LOCK
393  LockScourceInfos();
394 
395  while(true)
396  {
397  //If index is too large, loop back to zero
398  if(aCurrentSetIndex >= mActiveSourceInfos.size())
399  aCurrentSetIndex = 0;
400 
401  if(aCurrentSetIndex == aInputSetIndex)
402  {
403  //every source is nullptr, or there is only 1 open source
404  auto sInfo = mActiveSourceInfos[aCurrentSetIndex];
405  mScourceInfosLock = false; //UNLOCK
406  return sInfo;
407  }
408 
409  //If null, don't grab
410  if(mActiveSourceInfos[aCurrentSetIndex] == nullptr)
411  aCurrentSetIndex++;
412  else
413  break;
414  }
415  auto sInfo = mActiveSourceInfos[aCurrentSetIndex];
416 
417  //UNLOCK
418  mScourceInfosLock = false;
419 
420  aInputSetIndex = aCurrentSetIndex;
421  return sInfo;
422 }

425 // RegisterSourceFinished
426 //---------------------------------
427 JThreadManager::JEventSourceInfo* JThreadManager::RegisterSourceFinished(const JEventSource* aFinishedEventSource, std::size_t& aQueueSetIndex)
428 {
429  //Open the next source (if not done already)
430  //If two threads reach here at the same time, only one will open the source.
431  //The other thread will update to the new source later below
432  JEventSource::RETURN_STATUS sStatus = JEventSource::RETURN_STATUS::kUNKNOWN;
433  JEventSource* sNextSource = nullptr;
434  std::tie(sStatus, sNextSource) = mEventSourceManager->OpenNext(aFinishedEventSource);
435 
436  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID()
437  << " JThreadManager::RegisterSourceFinished(): Source finished, next source = "
438  << sNextSource << LOG_END;
439 
440  //LOCK
441  LockScourceInfos();
442 
443  if(sNextSource != nullptr)
444  {
445  //We have just opened a new event source
446  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID()
447  << " JThreadManager::RegisterSourceFinished(): New source opened" << LOG_END;
448 
449  //Retire the current queue set, insert a new one in its place
450  mActiveSourceInfos[aQueueSetIndex]->mQueueSet->FinishedWithQueues();
451  mRetiredSourceInfos.push_back(mActiveSourceInfos[aQueueSetIndex]);
452  auto sInfo = new JEventSourceInfo(sNextSource, MakeQueueSet(sNextSource));
453  mActiveSourceInfos[aQueueSetIndex] = sInfo;
454 
455  //Unlock and return
456  mScourceInfosLock = false;
457  return sInfo;
458  }
459 
460  //OK, the "next" source is null. Either another thread opened a new source, or there are no more sources to open.
461  if(sStatus == JEventSource::RETURN_STATUS::kUNKNOWN)
462  {
463  //OK, we're not quite sure what happened.
464  //When trying to open the next source, we searched for the old one but it was already registered as exhausted.
465  //So either:
466  //Another thread opened a new source, and installed it into mActiveQueueSets at the same slot
467  //There were no other sources to open, and installed nullptr in its place
468  auto sInfo = mActiveSourceInfos[aQueueSetIndex];
469  if(sInfo != nullptr)
470  {
471  //Another thread opened a new source, and installed it into mActiveQueueSets at the same slot
472  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID()
473  << " JThreadManager::RegisterSourceFinished(): Another thread opened a new source" << LOG_END;
474 
475  //Get and return the new queue set and source (don't forget to unlock!)
476  mScourceInfosLock = false;
477  return sInfo;
478  }
479 
480  //There were no new sources to open, so the other thread retired the queue set and put nullptr in its place
481 
482  //If done, tell all threads to end (if not done so already)
483  //If not, rotate this thread to the next open source and return the next queue set
484  return CheckAllSourcesDone(aQueueSetIndex);
485  }
486 
487  //We removed the previous source from the active list (first one to get there), and there are no new sources to open
488  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID()
489  << " JThreadManager::RegisterSourceFinished(): No new sources to open" << LOG_END;
490 
491  //Retire the current source info and remove it from the active queue
492  //Note that we can't actually erase it: It would invalidate the queue set indices of other threads
493  //Instead, we replace it with nullptr
494  if(mActiveSourceInfos[aQueueSetIndex] != nullptr) //if not done already by another thread
495  {
496  mRetiredSourceInfos.push_back(mActiveSourceInfos[aQueueSetIndex]);
497  mActiveSourceInfos[aQueueSetIndex] = nullptr;
498  }
499 
500  //If done, tell all threads to end
501  //If not, rotate this thread to the next open source and return the next queue set
502  return CheckAllSourcesDone(aQueueSetIndex);
503 }

505 //---------------------------------
506 // CheckAllSourcesDone
507 //---------------------------------
508 JThreadManager::JEventSourceInfo* JThreadManager::CheckAllSourcesDone(std::size_t& aQueueSetIndex)
509 {
510  //Only called by RegisterSourceFinished, and already within a lock on mScourceInfosLock
511  //This is only called once there are no more files to open, and we want to see if all jobs are done.
512  //If done, tell all threads to end
513  //If not, rotate this thread to the next open source and return the next queue set
514 
515  //Check if all sources are done
516  auto sNullChecker = [](const JEventSourceInfo* sInfo) -> bool {return (sInfo == nullptr);};
517  if(std::all_of(std::begin(mActiveSourceInfos), std::end(mActiveSourceInfos), sNullChecker))
518  {
519  //The last event source is done.
520  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID()
521  << " JThreadManager::CheckAllSourcesDone(): All tasks from all event sources are done" << LOG_END;
522 
523  //UNLOCK
524  mScourceInfosLock = false;
525 
526  //Tell all threads to end.
527  EndThreads();
528  return nullptr;
529  }
530 
531  //Not all sources are done yet.
532  //Rotate this thread to the next open/task-processing source.
533  do
534  {
535  aQueueSetIndex++;
536  if(aQueueSetIndex >= mActiveSourceInfos.size())
537  aQueueSetIndex = 0;
538  }
539  while(mActiveSourceInfos[aQueueSetIndex] == nullptr);
540 
541  //Get and return the next queue set and source (don't forget to unlock!)
542  auto sInfo = mActiveSourceInfos[aQueueSetIndex];
543  mScourceInfosLock = false;
544  return sInfo;
545 }

580 //---------------------------------
581 // PrepareEventTask
582 //---------------------------------
583 void JThreadManager::PrepareEventTask(const std::shared_ptr<JTaskBase>& aTask, const JEventSourceInfo* aSourceInfo) const
584 {
585  //If from the event queue, add factories to the event (JFactorySet)
586  //We don't add them earlier (e.g. in the event source) because it would take too much memory
587  //E.g. if 200 events in the queue, we'd have 200 factories of each type
588  //Instead, only the events that are actively being analyzed have factories
589  //Once the JEvent goes out of scope and releases the JFactorySet, it returns to the pool
590 
591  //OK, this is a bit ugly
592  //The problem is that JTask holds a shared_ptr of const JEvent*, and we need to modify it.
593  //Why not just hold a non-const shared_ptr?
594  //Because we don't want the user calling things like JEvent::SetEventNumber() in their factories.
595 
596  //So what do we do? The dreaded const_cast.
597  auto sEvent = const_cast<JEvent*>(aTask->GetEvent());
598 
599  //Is there something else we can do instead?
600  //We could pass the JFactorySet alongside the JEvent instead of as a member of it.
601  //But the user is not intended to ever directly interact with the JFactorySet
602  //And we'll also have to deal with this again for sending along barrier-event data.
603  //This ends up requiring a lot of arguments to factory/processor/task methods.
604 
605  //Now, let's set the JFactorySet:
606  //By having the JEvent Release() function recycle the JFactorySet, it doesn't have to be shared_ptr
607  //If it's shared_ptr: more (slow) atomic operations
608  //Just make the JFactorySet a member and cheat a little bit. It's a controlled environment.
609  sEvent->SetFactorySet(mApplication->GetFactorySet());
610 }

644  //Function does not return until all tasks are finished.
645  //Function ASSUMES all tasks from the same event source!!!
646 
647  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID()
648  << " JThreadManager::SubmitTasks(): Submit " << aTasks.size() << " tasks to queue type "
649  << static_cast<std::underlying_type<JQueueSet::JQueueType>::type>(aQueueType)
650  << ", name = " << aQueueName << LOG_END;
651 
652  if(aTasks.empty())
653  return;
654 
655  //Get queue
656  auto sEventSource = aTasks[0]->GetEvent()->GetEventSource();
657  auto sSourceInfo = GetEventSourceInfo(sEventSource);
658  auto sQueueSet = sSourceInfo->mQueueSet;
659  auto sQueue = sQueueSet->GetQueue(aQueueType, aQueueName);
660 
661  //Submit tasks
662  SubmitTasks(aTasks, sSourceInfo, sQueue, aQueueType);
663 
664  //While waiting for results, execute tasks from the queue we submitted to
665  DoWorkWhileWaitingForTasks(aTasks, sSourceInfo, sQueue, aQueueType);

669 // SubmitTasks
670 //---------------------------------
671 void JThreadManager::SubmitTasks(const std::vector<std::shared_ptr<JTaskBase>>& aTasks, JEventSourceInfo* aSourceInfo, JQueue* aQueue, JQueueSet::JQueueType aQueueType)
672 {
673  //You would think submitting tasks would be easy.
674  //However, the task JQueue can fill up, and we may not be able to put all of our tasks in.
675  //If that happens, we will execute one of the tasks in the queue before trying to add more.
676  //This function returns once all tasks are submitted.
677 
678  //Prepare iterators
679  auto sIterator = std::begin(aTasks);
680  auto sEnd = std::end(aTasks);
681 
682  //Loop over tasks and submit them
683  while(sIterator != sEnd)
684  {
685  //Try to submit task
686  if(aQueue->AddTask(*sIterator) != JQueue::Flags_t::kNO_ERROR)
687  {
688  //Failed (queue probably full): Execute a task in the meantime
689  ExecuteTask(*sIterator, aSourceInfo, aQueueType);
690  }
691  sIterator++; //advance to the next task
692  }
693 
694  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID() << " JThreadManager::SubmitTasks(): Tasks submitted." << LOG_END;
695 }

700 void JThreadManager::DoWorkWhileWaiting(const std::atomic<bool>& aWaitFlag, const JEventSource* aEventSource, JQueueSet::JQueueType aQueueType, const std::string& aQueueName)
701 {
702  //While waiting (while flag is true), execute tasks in the specified queue.
703  //Function does not return until the wait flag is false, which can only be changed by another thread.
704  //This is typically called during JEvent::Get(), if the objects haven't been created yet.
705  //In this case, one thread holds the input wait lock while creating them,
706  //and the other thread comes here and executes other tasks until the first thread finishes.
707 
708  //The thread holding the wait lock may be submitting its own tasks (and need them to finish first),
709  //so waiting threads should execute tasks from that queue to minimize the wait time.
710 
711  //Get queue for executing tasks
712  auto sSourceInfo = GetEventSourceInfo(aEventSource);
713  auto sQueueSet = sSourceInfo->mQueueSet;
714  auto sQueue = sQueueSet->GetQueue(aQueueType, aQueueName);
715 
716  //Build is-it-done checker
717  auto sWaitFunction = [&aWaitFlag](void) -> bool {return aWaitFlag;};
718 
719  //Execute tasks while waiting
720  DoWorkWhileWaitingForTasks(sWaitFunction, sSourceInfo, sQueue, aQueueType);
721 }

744 // DoWorkWhileWaitingForTasks
745 //---------------------------------
746 void JThreadManager::DoWorkWhileWaitingForTasks(const std::vector<std::shared_ptr<JTaskBase>>& aSubmittedTasks, JEventSourceInfo* aSourceInfo, JQueue* aQueue, JQueueSet::JQueueType aQueueType)
747 {
748  //The passed-in tasks have already been submitted (e.g. via SubmitTasks or SubmitAsyncTasks).
749  //While waiting for those to finish, we get tasks from the queues and execute those.
750  //The queue arguments should be those for the queue we just submitted the tasks to.
751  //That way, this thread will help do the work that was just submitted.
752 
753  //Is-it-done checker
754  auto sTaskCompleteChecker = [](const std::shared_ptr<JTaskBase>& sTask) -> bool {return sTask->IsFinished();};
755 
756  //Wait if all tasks aren't finished
757  //Pass in iterators instead of vector to avoid copying the vector (and ref counting)
758  auto sBegin = std::begin(aSubmittedTasks);
759  auto sEnd = std::end(aSubmittedTasks);
760  auto sWaitFunction = [sBegin, sEnd, sTaskCompleteChecker](void) -> bool {
761  return !std::all_of(sBegin, sEnd, sTaskCompleteChecker);};
762 
763  //Execute tasks while waiting
764  DoWorkWhileWaitingForTasks(sWaitFunction, aSourceInfo, aQueue, aQueueType);
765 }

770 void JThreadManager::DoWorkWhileWaitingForTasks(const std::function<bool(void)>& aWaitFunction, JEventSourceInfo* aSourceInfo, JQueue* aQueue, JQueueSet::JQueueType aQueueType)
771 {
772  //Won't return until aWaitFunction returns true.
773  //In the meantime, executes tasks (or sleeps if none available)
774  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID() << " JThreadManager::DoWorkWhileWaitingForTasks()" << LOG_END;
775 
776  while(aWaitFunction())
777  {
778  //Get task and execute it
779  auto sQueueType = JQueueSet::JQueueType::Events;
780  auto sTask = std::shared_ptr<JTaskBase>(nullptr);
781  std::tie(sQueueType, sTask) = GetTask(aSourceInfo->mQueueSet, aQueue, aQueueType);
782  if(sTask != nullptr)
783  ExecuteTask(sTask, aSourceInfo, sQueueType);
784  else
785  std::this_thread::sleep_for(mSleepTime); //Sleep a minimal amount.
786  }
787 
788  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID() << " JThreadManager::DoWorkWhileWaitingForTasks(): Done waiting." << LOG_END;
789 }

791 //---------------------------------
792 // GetTask
793 //---------------------------------
794 std::pair<JQueueSet::JQueueType, std::shared_ptr<JTaskBase>> JThreadManager::GetTask(JQueueSet* aQueueSet, JQueue* aQueue, JQueueSet::JQueueType aQueueType) const
795 {
796  //This is called by threads who have just submitted (or are waiting on) tasks to complete,
797  //and want work to do in the meantime.
798 
799  //This prefers getting tasks from the queue specified by the input arguments,
800  //then searches other queues if none are there.
801 
802  //Prefer executing tasks from the input queue
803  auto sTask = aQueue->GetTask();
804  if(sTask != nullptr)
805  return std::make_pair(aQueueType, sTask);
806 
807  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID() << " JThreadManager::GetTask(): Queue empty." << LOG_END;
808 
809  //Next, prefer executing tasks from the same event source (input queue set)
810  JQueueSet::JQueueType sQueueType;
811  std::tie(sQueueType, sTask) = aQueueSet->GetTask();
812  if(sTask != nullptr)
813  return std::make_pair(sQueueType, sTask);
814 
815  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID() << " JThreadManager::GetTask(): Event source tasks empty." << LOG_END;
816 
817  //Finally, loop over all active queue sets (different event sources)
818  LockScourceInfos();
819  for(auto sQueueSetInfo : mActiveSourceInfos)
820  {
821  if(sQueueSetInfo == nullptr)
822  continue;
823  auto sQueueSet = sQueueSetInfo->mQueueSet;
824  if(sQueueSet == aQueueSet)
825  continue; //already checked that one
826  std::tie(sQueueType, sTask) = aQueueSet->GetTask();
827  if(sTask != nullptr)
828  {
829  mScourceInfosLock = false; //UNLOCK
830  return std::make_pair(sQueueType, sTask);
831  }
832  }
833 
834  LOG_TRACE(mLogger, mDebugLevel) << "Thread " << JTHREAD->GetThreadID() << " JThreadManager::GetTask(): No tasks." << LOG_END;
835 
836  //No tasks in any queues, return nullptr
837  mScourceInfosLock = false; //UNLOCK
838  return std::make_pair(JQueueSet::JQueueType::Events, std::shared_ptr<JTaskBase>(nullptr));
839 }

842 // SetThreadAffinity
843 //---------------------------------
844 void JThreadManager::SetThreadAffinity(int affinity_algorithm)
845 {
846  /// Set the affinity of each thread. This will force each JThread
847  /// to be assigned to a specific HW thread. At this point, it just
848  /// assigns them sequentially. More sophisticated assignments should
849  /// be done by the user by getting a list of JThreads and then getting
850  /// a pointer to the std::thread object via the GetThread method.
851  ///
852  /// Note that setting the thread affinity is not something that can be
853  /// done through the C++ standard. It is done here only for pthreads
854  /// which is the underlying thread package for Linux and Mac OS X
855  /// (at least at the moment).
856 
857  // The default algorithm does not set the affinity at all
858  if( affinity_algorithm==0 ){
859  LOG_TRACE(mLogger, mDebugLevel) << "JThreadManager::SetThreadAffinity(): Thread affinity not set." << LOG_END;
860  return;
861  }
862 
863  if( typeid(std::thread::native_handle_type) != typeid(pthread_t) ){
864  LOG_WARN(mLogger) << "AFFINITY is set, but thread system is not pthreads." << LOG_END;
865  LOG_WARN(mLogger) << " Thread affinity will not be set by JANA." << LOG_END;
866  return;
867  }
868 
869  LOG_TRACE(mLogger, mLogTarget) << "JThreadManager::SetThreadAffinity(): Setting affinity for all threads using algorithm " << affinity_algorithm << LOG_END;
870 
871  uint32_t ithread = 0;
872  uint32_t ncores = GetNcores();
873  for( auto jt : mThreads ){
874  pthread_t t = jt->GetThread()->native_handle();
875 
876  uint32_t icpu = ithread;
877  switch( affinity_algorithm ){
878  case 1:
879  // This just assigns threads in numerical order.
880  // This is probably the best option for most
881  // production jobs.
882  break;
883 
884  case 2:
885  // This algorithm places subsequent threads ncores/2 apart
886  // which generally will pair them up on single cores.
887  // This is good if the threads benefit from sharing the L1
888  // and L2 cache. It is not good if they don't since they
889  // will likely compete for the core itself.
890  icpu = ithread/2;
891  icpu += ((ithread%2)*ncores/2);
892  icpu %= ncores;
893  break;
894  default:
895  LOG_FATAL(mLogger) << "Unknown affinity algorithm " << affinity_algorithm << LOG_END;
896  exit( -1 );
897  break;
898  }
899  ithread++;
900 
901 #ifdef __APPLE__
902  // Mac OS X
903  thread_affinity_policy_data_t policy = { (int) icpu };
904  thread_port_t mach_thread = pthread_mach_thread_np(t);
905  thread_policy_set(mach_thread, THREAD_AFFINITY_POLICY,
906  (thread_policy_t)&policy,
907  THREAD_AFFINITY_POLICY_COUNT);
908 
909  LOG_DEBUG(mLogger) << "CPU=" << mApplication->GetCPU()
910  << " (mach_thread="<< mach_thread
911  << ", icpu=" << icpu << ")" << LOG_END;
912 #else
913  // Linux
914  cpu_set_t cpuset;
915  CPU_ZERO(&cpuset);
916  CPU_SET(icpu, &cpuset);
917  int rc = pthread_setaffinity_np(t, sizeof(cpu_set_t), &cpuset);
918  if (rc != 0) {
919  LOG_ERROR(mLogger) << "pthread_setaffinity_np returned "
920  << rc << " for thread " << ithread
921  << LOG_END;
922  }
923 
924 #endif
925  }
926 }

953 //---------------------------------
954 // StopThreads
955 //---------------------------------
956 void JThreadManager::StopThreads(bool wait_until_idle)
957 {
958  /// Stop all threads from processing. This will allow the threads to complete
959  /// their current task before dropping into the idle state. This will call
960  /// the Stop() method for all JThread objects.
961  ///
962  /// Note that JThread::Stop() will be called with no arguments first which should
963  /// flag the thread to stop, but return right away. If the wait_until_idle flag
964  /// is set, then this will call JThread::Stop a second time, passing an argument
965  /// of "true" which will block until the thread is in the idle state. By calling
966  /// the Stop method twice, it will more quickly get to the state where all threads
967  /// are idle.
968  ///
969  /// Note also that if RunThreads() is called (or JThread::Run()) before a thread has
970  /// gone into the idle state then that will effectively cancel this call and it may
971  /// return with one of more threads never having gone into the idle state. In other
972  /// words, calling this with wait_until_idle==true does no necessarily guarrantee that
973  /// all threads are idle upon return.
974 
975  for(auto sThread : mThreads) sThread->Stop();
976 
977  if( wait_until_idle ) for(auto sThread : mThreads) sThread->Stop( true );
978 }
