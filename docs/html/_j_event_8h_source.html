179  // This may have been something Paul was working on but didn't quite complete.

184 // if(!mIsBarrierEvent && (mLatestBarrierEvent != nullptr))

200 // if(sFactory->GetCreated())
201 // return sFactory->Get();
202 // }

205  //First get the factory
206  auto sFactoryBase = mFactorySet->GetFactory(std::type_index(typeid(DataType)), aTag);
207  if(sFactoryBase == nullptr)
208  {
209  //Uh oh, No factory exists for this type.
210  jerr << "ERROR: No factory found for type = " << GetDemangledName<DataType>() << ", tag = \"" << aTag << "\"\n";
211  japp->SetExitCode(-1);
212  japp->Quit();
213  return {};
214  }
215  auto sFactory = static_cast<JFactoryT<DataType>*>(sFactoryBase);
216 
217  //If objects previously created, just return them
218  if(sFactory->GetCreated()) return sFactory->Get();
219 
220  // Objects are not already created so we may need to create them.
221  // Ensure the Init method has been called for the factory.
222  std::call_once(sFactory->init_flag, &JFactory::Init, sFactory);
223 
224  //Attempt to acquire the "creating" lock for the factory
225  if(!sFactory->AcquireCreatingLock())
226  {
227  if(mDebugLevel >= 10)
228  JLog() << "Thread " << JTHREAD->GetThreadID() << " JEvent::Get(): Another thread is creating objects, do work while waiting.\n" << JLogEnd();
229

232  mThreadManager->DoWorkWhileWaiting(sFactory->GetCreatingLock(), mEventSource);
233 
234  //It's done, return the results
235  return sFactory->Get();
236  }
237 
238  if(mDebugLevel >= 10)
239  JLog() << "Thread " << JTHREAD->GetThreadID() << " JEvent::Get(): Create lock acquired.\n" << JLogEnd();
240 
241  //If we throw exception within here we may not release the lock, unless we do try/catch:
242  try
243  {
244  //Lock acquired. First check to see if they were created since the last check.
245  if(sFactory->GetCreated())
246  {
247  if(mDebugLevel >= 10)
248  JLog() << "Thread " << JTHREAD->GetThreadID() << " JEvent::Get(): Objects created in meantime.\n" << JLogEnd();
249  sFactory->ReleaseCreatingLock();
250  return sFactory->Get();
251  }
252 
253  //Not yet: We need to create the objects.
254  //First try to get from the event source
255  if(mDebugLevel >= 10)
256  JLog() << "Thread " << JTHREAD->GetThreadID() << " JEvent::Get(): Try to get " << GetDemangledName<DataType>() << " (tag = " << aTag << ") objects from JEventSource.\n" << JLogEnd();
257  auto sSharedThis = this->shared_from_this();
258  if (mEventSource != nullptr &&
259  mEventSource->GetObjects(sSharedThis, static_cast<JFactory*>(sFactory)))
260  {
261  if(mDebugLevel >= 10)
262  JLog() << "Thread " << JTHREAD->GetThreadID() << " JEvent::Get(): " << GetDemangledName<DataType>() << " (tag = " << aTag << ") retrieved from JEventSource.\n" << JLogEnd();
263  sFactory->SetCreated(true);
264  sFactory->ReleaseCreatingLock();
265  return sFactory->Get();
266  }
267 
268  //Not in the file: have the factory make them
269  //First compare current run # to previous run. If different, call ChangeRun()
270  if(sFactory->GetPreviousRunNumber() != mRunNumber)
271  {
272  if(mDebugLevel >= 10)
273  JLog() << "Thread " << JTHREAD->GetThreadID() << " JEvent::Get(): Change run.\n" << JLogEnd();
274  sFactory->ChangeRun(sSharedThis);
275  sFactory->SetPreviousRunNumber(mRunNumber);
276  }
277 
278  //Create the objects
279  if(mDebugLevel >= 10)
280  JLog() << "Thread " << JTHREAD->GetThreadID() << " JEvent::Get(): Create " << GetDemangledName<DataType>() << " (tag = " << aTag << ") with factory.\n" << JLogEnd();
281  sFactory->Process(sSharedThis);
282  sFactory->SetCreated(true);
283  sFactory->ReleaseCreatingLock();
284  }
285  catch(...)
286  {
287  //Catch the exception, unlock, rethrow
288  auto sException = std::current_exception();
289  sFactory->ReleaseCreatingLock();
290  std::rethrow_exception(sException);
291  }
292 
293  //Get the object iterators
294  auto sIteratorPair = sFactory->Get();
295  if(mDebugLevel > 0)
296  JLog() << "Thread " << JTHREAD->GetThreadID() << " JEvent::Get(): Getting " << std::distance(sIteratorPair.first, sIteratorPair.second) << " " << GetDemangledName<DataType>() << " objects, tag = " << aTag << ".\n" << JLogEnd();
297 
298  //Return the objects
299  return sIteratorPair;
300 }
