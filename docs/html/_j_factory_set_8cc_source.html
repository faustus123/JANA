59 JFactorySet::JFactorySet(const std::vector<JFactoryGenerator*>& aFactoryGenerators)
60 {
61  //Add all factories from all factory generators
62  for(auto sGenerator : aFactoryGenerators){
63 
64  // Generate the factories into a temporary JFactorySet.
65  JFactorySet myset;
66  sGenerator->GenerateFactories( &myset );
67 
68  // Merge factories from temporary JFactorySet into this one. Any that
69  // already exist here will leave the duplicates in the temporary set
70  // where they will be destroyed by its destructor as it falls out of scope.
71  Merge( myset );
72  }
73 }

86 //---------------------------------
87 bool JFactorySet::Add(JFactory* aFactory)
88 {
89  /// Add a factory to this JFactorySet. The caller should be careful when
90  /// using the factory pointer after calling this since the factory may be deleted
91  /// during this call. This will happen if a factory of the same type and tag
92  /// already exists in this set. A return value of "true" indicates that the
93  /// factory was successfully added and that the given pointer is still valid
94  /// (until this set is destroyed). A return value of "false" indicates that
95  /// this factory was a duplicate and therefore the factory object was destroyed.
96  /// In all cases, this assumes ownership of the factory object.
97 
98  auto sKey = std::make_pair( aFactory->GetObjectType(), aFactory->GetTag() );
99  auto res = mFactories.emplace(sKey, static_cast<JFactory*>(aFactory));
100  if( res.second ) return true; // factory successfully added
101 
102  // factory is duplicate. Destroy it and inform caller by returning false.
103  delete aFactory;
104 
105  return false;
106 }

119 // Merge
120 //---------------------------------
121 void JFactorySet::Merge(JFactorySet &aFactorySet)
122 {
123  /// Merge any factories in the specified JFactorySet into this
124  /// one. Any factories which don't have the same type and tag as one
125  /// already in this set will be transferred and this JFactorySet
126  /// will take ownership of them. Ones that have a type and tag
127  /// that matches one already in this set will be left in the
128  /// original JFactorySet. Thus, all factories left in the JFactorySet
129  /// passed into this method upon return from it can be considered
130  /// duplicates. It will be left to the caller to delete those.
131 
132  JFactorySet tmpSet; // keep track of duplicates to copy back into aFactorySet
133  for( auto f : aFactorySet.mFactories ){
134  if( ! mFactories.insert( f ).second ) {
135  // duplicate. Record so we can send back to caller
136  tmpSet.mFactories[f.first] = f.second;
137  }
138  }
139 
140  // Copy duplicates back to aFactorySet
141  aFactorySet.mFactories.swap( tmpSet.mFactories );
142  tmpSet.mFactories.clear(); // prevent ~JFactorySet from deleting any factories
143 }

148 void JFactorySet::Print(void) const
149 {
150  size_t max_len = 0;
151  for( auto p: mFactories ){
152  auto len = p.second->GetName().length();
153  if( len > max_len ) max_len = len;
154  }
155 
156  max_len += 4;
157  for( auto p: mFactories ){
158  auto name = p.second->GetName();
159  auto tag = p.second->GetTag();
160 
161  std::cout << std::string( max_len-name.length(), ' ') + name;
162  if(!tag.empty()) std::cout << ":" << tag;
163  std::cout << std::endl;
164  }
165 }
