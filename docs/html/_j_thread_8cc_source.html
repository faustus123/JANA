95 uint64_t JThread::GetNumEventsProcessed(void)
96 {
97  /// Get total number of events processed by this thread. This
98  /// returns a total of all "events" from all queues. Since the
99  /// nature of events from different queues differs (e.g. one
100  /// event in queue "A" may turn into 100 events in queue "B")
101  /// this value should be used with caution. See the other form
102  /// GetNeventsProcessed(map<string,uint64_t> &) which returns
103  /// counts for individual queues for a perhaps more useful breakdown.
104 
105  std::map<std::string,uint64_t> Nevents;
106  GetNumEventsProcessed(Nevents);
107 
108  uint64_t Ntot = 0;
109  for(auto p : Nevents) Ntot += p.second;
110 
111  return Ntot;
112 }

150 void JThread::Join(void)
151 {
152  /// Join this thread. If the thread is not already in the ended
153  /// state then this will call End() and wait for it to do so
154  /// before calling join. This should generally only be called
155  /// from a method JThreadManager.
156  if(_isjoined) return;
157  if( mRunStateTarget != kRUN_STATE_ENDED ) End();
158  while( mRunState != kRUN_STATE_ENDED ) std::this_thread::sleep_for( std::chrono::microseconds(100) );
159  _thread->join();
160  _isjoined = true;
161  delete _thread;
162  _thread = nullptr;
163 }

236 void JThread::Loop(void)
237 {
238  // Set thread_local global variable
239  JTHREAD = this;
240 
241  //Set logger
242  mLogger = new JLog(0); //std::cout
243 
244  /// Loop continuously, processing events
245  try{
246  while( mRunStateTarget != kRUN_STATE_ENDED )
247  {
248  // If specified, go into idle state
249  if( mRunStateTarget == kRUN_STATE_IDLE ) mRunState = kRUN_STATE_IDLE;
250 
251  // If not running, sleep and loop again
252  if(mRunState != kRUN_STATE_RUNNING)
253  {
254  std::this_thread::sleep_for(mSleepTime); //Sleep a minimal amount.
255  continue;
256  }
257 
258  //Check if not enough event-tasks queued
259  if(CheckEventQueue())
260  {
261  //Process-event task is submitted, redo the loop in case we want to buffer more
262  continue;
263  }
264 
265  // Grab the next task
266  if(mDebugLevel >= 50) *mLogger << "Thread " << mThreadID << " JThread::Loop(): Grab task\n" << JLogEnd();
267  auto sQueueType = JQueueSet::JQueueType::Events;
268  auto sTask = std::shared_ptr<JTaskBase>(nullptr);
269  std::tie(sQueueType, sTask) = mEventSourceInfo->mQueueSet->GetTask();
270  if(mDebugLevel >= 50)
271  *mLogger << "Thread " << mThreadID << " JThread::Loop(): Task pointer: " << sTask << "\n" << JLogEnd();
272 
273  //Do we have a task?
274  if(sTask == nullptr)
275  {
276  if(!HandleNullTask()) break; //No more tasks, end the loop
277  continue;
278  }
279 
280  //Execute task
281  mThreadManager->ExecuteTask(sTask, mEventSourceInfo, sQueueType);
282 
283  //Task complete. If this was an event task, rotate to next open file (if desired)
284  //Don't rotate if it was a subtask or output task, because many of these may have submitted at once and we want to finish those first
285  if(mRotateEventSources)
286  {
287  mFullRotationCheckIndex = mQueueSetIndex; //reset for full-rotation-with-no-action check
288  if(sQueueType == JQueueSet::JQueueType::Events)
289  {
290  mEventSourceInfo = mThreadManager->GetNextSourceQueues(mQueueSetIndex);
291  if(mEventSourceInfo == nullptr)
292  {
293  mRunStateTarget = kRUN_STATE_ENDED;
294  continue; //Somehow down to no tasks left: We're done processing
295  }
296  if(mDebugLevel >= 20)
297  *mLogger << "Thread " << mThreadID << " JThread::Loop(): Rotated sources to: " << mQueueSetIndex << ", rotation check = " << mFullRotationCheckIndex << "\n" << JLogEnd();
298  mEventQueue = mEventSourceInfo->mQueueSet->GetQueue(JQueueSet::JQueueType::Events);
299  mSourceEmpty = false;
300  }
301  }
302  }
303  }catch(JException &e){
304  jerr << "** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** " << std::endl;
305  jerr << "Caught JException: " << e.GetMessage() << std::endl;
306  jerr << "** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** ** " << std::endl;
307  }
308 
309  // Set flag that we're done just before exiting thread
310  mRunState = kRUN_STATE_ENDED;
311 }

314 // CheckEventQueue
315 //---------------------------------
316 bool JThread::CheckEventQueue(void)
317 {
318  //Returns true if the event-task buffer is too low and we read-in/prepared new events
319  //Otherwise (buffer is high enough or there are no events) return false
320 
321  if(mDebugLevel >= 50)
322  *mLogger << "Thread " << mThreadID << " JThread::CheckEventQueue(): Is-empty, enough buffered: " << mSourceEmpty << ", " << mEventQueue->AreEnoughTasksBuffered() << "\n" << JLogEnd();
323  if(mSourceEmpty || mEventQueue->AreEnoughTasksBuffered())
324  return false; //Didn't buffer more events, just execute the next available task
325 
326  //Figure out how many event (tasks) to get
327  auto sNumEventsToGetRange = mEventSourceInfo->mEventSource->GetNumEventsToGetAtOnce();
328  auto sNumEventsRangeSize = sNumEventsToGetRange.second - sNumEventsToGetRange.first;
329 
330  //The default is to get range max, unless the range size is non-zero or there are many threads
331  //If only one thread: Get the max: maximize cache coherency, keep the disk hot
332  //Or else we'll just have to keep getting from file one at a time over and over again
333  auto sNumEventsToGet = sNumEventsToGetRange.second;
334  if((sNumEventsRangeSize > 0) && (mThreadManager->GetNJThreads() > 1))
335  {
336  //The # of events we'll get is based on how full the buffer is.
337  //If the buffer is nearly empty, we'll get very few events.
338  //Why? So that way the other threads won't have to wait as long for something to do
339  //If the buffer is nearly full, we'll get a lot of events
340  //Why? Because the other threads have plenty of work to keep them busy.
341  //We only want one thread worried about reading from the source.
342  //Minimizes contention, keeps the disk hot, maximizes cache coherency
343  //Note that the buffer shouldn't be thought of as a MAX, but as a MIN.
344  //Thus we can skyrocket over it
345  auto sBufferFillFraction = double(mEventQueue->GetNumTasks()) / double(mEventQueue->GetTaskBufferSize());
346  if(sBufferFillFraction > 1.0)
347  sBufferFillFraction = 1.0;
348  sNumEventsToGet = sNumEventsToGetRange.first + sBufferFillFraction*(double(sNumEventsRangeSize));
349  }
350 
351  if(mDebugLevel >= 20)
352  *mLogger << "Thread " << mThreadID << " JThread::CheckEventQueue(): Get " << sNumEventsToGet << " process event tasks\n" << JLogEnd();
353 
354  //Get the next event(s) from the source, and get task(s) to process it/them (unless another thread has locked access)
355 // auto sTasksPair = mEventSourceInfo->mEventSource->GetProcessEventTasks(sNumEventsToGet);
356 // auto sReturnStatus = sTasksPair.second;
357  auto sEventTasks = mEventSourceInfo->mEventSource->GetProcessEventTasks(sNumEventsToGet);
358  if( !sEventTasks.empty() ){
359  if(mDebugLevel >= 40)
360  *mLogger << "Thread " << mThreadID << " JThread::CheckEventQueue(): Success, add task(s)\n" << JLogEnd();
361 
362  //Loop over tasks
363  for(auto& sEventTask : sEventTasks)
364  {
365  //Add this process-event task to the event queue.
366  //We have no where else to put this task if it doesn't fit in the queue.
367  //So just continually try to add it
368  if(mEventQueue->AddTask(std::move(sEventTask)) != JQueue::Flags_t::kNO_ERROR)
369  {
370  //Add failed, queue must be full.
371  //This (probably) shouldn't happen if the relationship between the
372  //queue size and the buffer size is reasonable.
373 
374  if(mDebugLevel >= 40) *mLogger << "Thread " << mThreadID << " JThread::CheckEventQueue(): Unable to add task, executing... \n" << JLogEnd();
375 
376  //Oh well. Just execute this task directly instead
377  //This is faster than pulling an event off the front of the queue
378  (*sEventTask)();
379  mEventQueue->AddTasksProcessedOutsideQueue(1);
380  }
381  }
382 
383  //Process-event task(s) submitted
384  return true;
385  }
386 
387  if(mDebugLevel >= 40) *mLogger << "Thread " << mThreadID << " JThread::CheckEventQueue(): No task(s) added\n" << JLogEnd();
388 
389  // No event tasks were read from the source
390  if( mEventSourceInfo->mEventSource->IsExhausted() ) mSourceEmpty = true;
391 
392  // We may get here for a variety of reasons. All of them though indicate
393  // that no events were obtained so return false to let the caller know that
394  // nothing was added to any queues.
395  return false;
396 }

399 // HandleNullTask
400 //---------------------------------
401 bool JThread::HandleNullTask(void)
402 {
403  //There are no tasks in the queue for this event source. Are we done with the source?
404  //We are not done with a source until all tasks referencing it have completed.
405 
406  //Just because there isn't a task in any of the queues doesn't mean we are done:
407  //A thread may have removed a task and be currently running it.
408  //Also, that thread may spawn a bunch of sub-tasks, so we still want all threads
409  //to be available to process them.
410 
411  //This returns false if there are no tasks left to process. Otherwise returns true.
412 
413  if(mDebugLevel >= 10)
414  *mLogger << "Thread " << mThreadID << " JThread::HandleNullTask(): Null task: is empty, # outstanding = " << mSourceEmpty << ", " << mEventSourceInfo->mEventSource->GetNumOutstandingEvents() << "\n" << JLogEnd();
415 
416  if(mSourceEmpty)
417  {
418  //So, how can we tell if all threads are done processing tasks from a given file?
419  //When all JEvent's associated with that file are destroyed/recycled.
420  //We track this by counting the number of open JEvent's in each JEventSource.
421 
422  auto sEventSource = mEventSourceInfo->mEventSource;
423  auto sNumOutstandingEvents = sEventSource->GetNumOutstandingEvents();
424  auto sEventsCheck = (sNumOutstandingEvents == 0);
425 
426  //The tricky part: There may be one event left, and it may be a barrier event.
427  //Even trickier: We may be still analyzing this event rather than finished with it.
428  //So, we are done if: 1 event left, 1 barrier event left, barrier shared_ptr use count = 1
429  auto sBarrierCheck = false;
430  if(!sEventsCheck) //don't bother checking if events check is already true
431  sBarrierCheck = (sNumOutstandingEvents == 1) && (sEventSource->GetNumOutstandingBarrierEvents() == 1) && (mEventQueue->GetLatestBarrierEventUseCount() == 1);
432 
433  if(sEventsCheck || sBarrierCheck)
434  {
435  //Tell the thread manager that the source is finished (in case it didn't know already)
436  //Use the existing queues for the next event source
437  //If all sources are done, then all tasks are done, and this call will tell the program to end.
438 
439  mEventSourceInfo = mThreadManager->RegisterSourceFinished(mEventSourceInfo->mEventSource, mQueueSetIndex);
440  if(mEventSourceInfo == nullptr)
441  return false; //No tasks left: We're done processing
442 
443  mEventQueue = mEventSourceInfo->mQueueSet->GetQueue(JQueueSet::JQueueType::Events);
444  mSourceEmpty = false;
445  mFullRotationCheckIndex = mQueueSetIndex; //reset for full-rotation-with-no-action check
446  return true;
447  }
448  }
449 
450  if(mRotateEventSources) //No tasks, try to rotate to a different input file
451  {
452  mEventSourceInfo = mThreadManager->GetNextSourceQueues(mQueueSetIndex);
453  if(mEventSourceInfo == nullptr)
454  return false; //Somehow in between, down to no tasks left: We're done processing
455  mEventQueue = mEventSourceInfo->mQueueSet->GetQueue(JQueueSet::JQueueType::Events);
456  mSourceEmpty = false;
457  if(mDebugLevel >= 20)
458  *mLogger << "Thread " << mThreadID << " JThread::HandleNullTask(): Rotated sources to " << mQueueSetIndex << ", rotation check = " << mFullRotationCheckIndex << "\n" << JLogEnd();
459 
460  //Check if we have rotated through all open event sources without doing anything
461  //If so, we are probably waiting for another thread to finish a task, or for more events to be ready.
462  if(mQueueSetIndex == mFullRotationCheckIndex) //yep
463  std::this_thread::sleep_for(mSleepTime); //Sleep a minimal amount.
464  }
465  else //No tasks, wait a bit for this event source to be ready
466  std::this_thread::sleep_for(mSleepTime); //Sleep a minimal amount.
467 
468  return true;
469 }

485 void JThread::Stop(bool wait_until_idle)
486 {
487  /// Stop the thread from processing events. The stop will occur
488  /// between events so if an event is currently being processed,
489  /// that will complete before the thread goes idle. If wait_until_idle
490  /// is true, then this will not return until the thread is no longer
491  /// in the "running" state and therefore no longer processing events.
492  /// This will usually be because the thread has gone into the idle
493  /// state, but may also be because the thread has been told to end
494  /// completely.
495  /// Use IsIdle() to check if the thread is in the idle
496  /// state.
497  mRunStateTarget = kRUN_STATE_IDLE;
498 
499  // The use of yield() here will (according to stack overflow) cause
500  // the core to be 100% busy while waiting for the state to change.
501  // That should only be for the duration of processing of the current
502  // event though and this method is expected to be used rarely so
503  // this should be OK.
504  if( wait_until_idle ){
505  while( (mRunState==kRUN_STATE_RUNNING) && (mRunStateTarget==kRUN_STATE_IDLE) ) std::this_thread::yield();
506  }
