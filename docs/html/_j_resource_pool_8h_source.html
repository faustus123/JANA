16 /****************************************************** OVERVIEW ******************************************************
17  *
18  * This class can be used as a wrapper to a (private) static resource pool that is accessible to all class instances.
19  * A pool counter keeps track of how many JResourcePool's there are for a given type.
20  * Once it drops to zero, all of the remaining objects are deleted.
21  *
22  ***************************************** DEFINITION AND USE: NON-SHARED_PTR'S ***************************************
23  *
24  * You can retrieve and recycle resources via the Get_Resource() and Recycle() functions.
25  * Be sure to recycle the memory once you are done with it or else it will leak.
26  *
27  ******************************************* DEFINITION AND USE: SHARED_PTR'S *****************************************
28  *
29  * You can retrieve shared_ptr's of the objects by calling the Get_SharedResource() method.
30  * The advantage of using shared_ptr's is that they automatically keep track of when they are out of scope.
31  * These shared_ptr's have been created with a JSharedPtrRecycler functor:
32  * Once the shared_ptr goes out of scope, the contained resource is automatically recycled back to the pool.
33  *
34  * The shared_ptr MUST NOT outlive the life of the JResourcePool!!! Or else it will crash.
35  *
36  **********************************************************************************************************************/

92  {
93  /// Returns the cache line size for the processor of the target platform.
94  /*! The cache line size is useful for creating a buffer to make sure that a variable accessed by multiple threads does not share the cache line it is on.

96  This blocks other threads from operating on the other data stored on the cache line. Note that it is also important to align the shared data as well.

374 template <typename DType>
375 void JResourcePool<DType>::Recycle_Resources_StaticPool(std::vector<DType*>& sResources)
376 {
377  if(dDebugLevel >= 10)
378  std::cout << "RECYCLE " << sResources.size() << " RESOURCES " << typeid(DType).name() << std::endl;
379 
380  std::size_t sFirstToCopyIndex = 0;
381  {
382  LockPool();
383 
384  auto sPotentialNewPoolSize = mResourcePool.size() + sResources.size();
385  if(sPotentialNewPoolSize > dMaxPoolSize) //we won't move all of the resources into the shared pool, as it would be too large: only move a subset
386  sFirstToCopyIndex = sResources.size() - (dMaxPoolSize - mResourcePool.size());
387 
388  std::copy(sResources.begin() + sFirstToCopyIndex, sResources.end(), std::back_inserter(mResourcePool));
389  dPoolLock = false; //Unlock
390  }
391 
392  if(dDebugLevel > 0)
393  {
394  std::cout << "PUT IN SHARED POOL " << typeid(DType).name() << ": " << sResources.size() - sFirstToCopyIndex << std::endl;
395  std::cout << "DELETING " << typeid(DType).name() << ": " << sFirstToCopyIndex << std::endl;
396  }
397 
398  //any resources that were not copied into the shared pool are deleted instead (too many)
399  auto Deleter = [](DType* sResource) -> void {delete sResource;};
400  std::for_each(sResources.begin(), sResources.begin() + sFirstToCopyIndex, Deleter);
401 
402  dObjectCounter -= sFirstToCopyIndex;
403  sResources.clear();
404 }
