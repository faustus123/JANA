119 def process_command( input ):
120  global CLI_ACTIVE, BANNER_ON, mode
121 
122  stripped = input.rstrip()
123  tokens = input.rstrip().split()
124  if len(tokens) < 1: return
125  cmd = tokens[0]
126  args = tokens[1:]
127 
128  print 'processing command: ' + ' '.join(tokens)
129  LEFT(1000)
130 
131  #--- banner
132  if cmd=='banner':
133  if len(args) == 0:
134  print 'command banner requires you to specify "on" or "off". (see help for details)'
135  elif args[0] == 'on' : BANNER_ON = True
136  elif args[0] == 'off': BANNER_ON = False
137  else: print 'command banner requires you to specify "on" or "off". (see help for details)'
138  #--- exit, quit
139  elif cmd=='exit' or cmd=='quit':
140  CLI_ACTIVE = False
141  jana.Quit()
142  #--- help
143  elif cmd=='help':
144  print_help()
145  #--- history
146  elif cmd=='history':
147  idx = 0
148  for h in history:
149  print '%3d %s\r' % (idx, h)
150  idx += 1
151  print '%3d %s\r' % (idx, ' '.join(tokens) ) # include this history command which will be added below
152  #--- nthreads
153  elif cmd=='nthreads':
154  if len(args) == 0:
155  print 'Number of JThreads: %d' % jana.GetNJThreads()
156  elif len(args)==1:
157  print 'Number of JThreads now at: %d' % jana.SetNJThreads( int(args[0]) )
158  else:
159  print 'njthreads takes either 0 or 1 argument (see help for details)'
160  #--- parameter
161  elif cmd=='parameter':
162  if len(args) == 0:
163  print 'command parameter requires arguments. (see help for details)'
164  elif args[0] == 'get':
165  if len(args)==2:
166  print '%s: %s' % (args[1], jana.GetParameterValue(args[1]))
167  else:
168  print 'command parameter get requires exactly 1 argument! (see help for details)'
169  elif args[0] == 'set':
170  if len(args)==3:
171  jana.SetParameter(args[1], args[2])
172  else:
173  print 'command parameter set requires exactly 2 arguments! (see help for details)'
174  elif args[0] == 'list':
175  if len(args)==1:
176  jana.PrintParameters(True)
177  else:
178  print 'command parameter list requires no arguments! (see help for details)'
179  #--- resume
180  elif cmd=='resume':
181  jana.Resume()
182  print 'event processing resumed'
183  #--- status
184  elif cmd=='status':
185  jana.PrintStatus()
186  print ''
187  #--- stop
188  elif cmd=='stop':
189  jana.Stop(True)
190  print 'event processing stopped after %d events (use "resume" to start again)' % jana.GetNeventsProcessed()
191  #--- unknown command
192  else:
193  print 'Unknown command: ' + cmd
194  return
195 
196  history.append( ' '.join(tokens) ) # add to history

215 def command_line():
216 
217  global CLI_ACTIVE, history, mode, BANNER_ON
218 
219  # Record tty settings so we can restore them when finished
220  # processing the CLI
221  mode = termios.tcgetattr(sys.stdin)
222 
223  # Wait for first event to process so all messages are printed
224  # before switching to "raw" mode which screws up newlines
225  while jana.GetNeventsProcessed()<1: time.sleep(1)
226 
227  # Set to raw mode so single characters get passed to us as they are typed
228  tty.setraw(sys.stdin)
229 
230  prompt = 'jana> '
231 
232  CLI_ACTIVE = True
233  BANNER_ON = True
234  history = []
235  last_draw_time = 0
236 
237  while CLI_ACTIVE and not jana.IsDrainingQueues(): # loop for each line
238  # Define data-model for an input-string with a cursor
239  input = ""
240  index = 0
241  input_save = ""
242  history_index = len(history)
243  sys.stdout.write(prompt)
244  sys.stdout.flush()
245  while CLI_ACTIVE and not jana.IsDrainingQueues(): # loop for each character
246 
247  # Redraw the screen periodically
248  if BANNER_ON and (time.time() - last_draw_time > 1):
249  draw_banner()
250  MOVETO(1,NROWS)
251  if index >= 0: RIGHT( index+len(prompt) ) ; sys.stdout.flush()
252  last_draw_time = time.time()
253 
254  # Wait for a character to be typed. 0.100 is timeout in seconds
255  # Doing this first allows us to check if the app is quitting
256  # even if the user is not typing so *maybe* we can restore the
257  # termios settings before the final report is printed. It also
258  # allows us to update the screen periodically when nothing is
259  # being typed.
260  i, o, e = select.select( [sys.stdin], [], [], 0.100 )
261  if not i: continue # nothing was typed
262 
263  char = ord(sys.stdin.read(1)) # read one char and get char code
264 
265  # Manage internal data-model
266  if char == 3: return # CTRL-C
267  elif char in {10, 13}: break # LF or CR
268  elif char == 9: # TAB
269  rindex = len(input) - index
270  input = tab_complete(input)
271  index = len(input) - rindex
272  elif 32 <= char <= 126: # Non-special character
273  input = input[:index] + chr(char) + input[index:]
274  index += 1
275  elif char == 127: # Delete/Backspace
276  input = input[:index-1] + input[index:]
277  index -= 1
278  elif char == 27: # Left or right arrow start
279  next1, next2 = ord(sys.stdin.read(1)), ord(sys.stdin.read(1))
280  if next1==91 and next2==68: index = max(0, index - 1) # Left arrow
281  if next1==91 and next2==67: index = min(len(input), index + 1) # Right arrow
282  if next1==91 and next2==66: # Down arrow
283  if history_index < (len(history)-1):
284  history_index += 1
285  input = history[ history_index ]
286  index = len(input)
287  elif history_index == (len(history)-1):
288  history_index += 1
289  input = input_save
290  index = len(input)
291 
292  if next1==91 and next2==65: # Up arrow
293  if history_index > 0:
294  if history_index == len(history): input_save = input
295  history_index -= 1
296  input = history[ history_index ]
297  index = len(input)
298 
299 # else:
300 # print '\nunknown char: %d\n' % int(char)
301 # sys.stdout.flush()
302 
303  # Print current input-string
304  MOVETO(1,NROWS)
305  CLEARLINE()
306  sys.stdout.write(prompt + syntax_highlight(input))
307  MOVETO(1,NROWS)
308  if index >= 0: RIGHT( index+len(prompt) )
309  sys.stdout.flush()
310 
311  # Process one input line
312  print ''
313  MOVETO(1,NROWS)
314  termios.tcsetattr(sys.stdin, termios.TCSAFLUSH, mode)
315  process_command( input )
316  tty.setraw(sys.stdin)
317  MOVETO(1,NROWS)
318  sys.stdout.flush()
319  input = ""
320  index = 0
321  last_draw_time = 0 # force banner to redraw on next iteration
322 
323  # Restore terminal settings
324  termios.tcsetattr(sys.stdin, termios.TCSAFLUSH, mode)
325  print '\nCLI exiting ...\n'
