145 //---------------------------------
146 // TestThread
147 //
148 // This method will be called as a separate thread to orchestrate
149 // the test depending on the testing mode.
150 //---------------------------------
151 void JTestMain::TestThread(void)
152 {
153  //- - - - - - - - - - - - - - - - - - - - - - - -
154  // Scaling test
155 
156  auto tm = mApp->GetJThreadManager();
157 
158  // Turn ticker off so we can better control the screen
159  mApp->SetTicker( false );
160 
161  // Wait for events to start flowing indicating the source is primed
162  for(int i=0; i<60; i++){
163  cout << "Waiting for event source to start producing ... rate: " << mApp->GetInstantaneousRate() << endl;
164  std::this_thread::sleep_for( std::chrono::milliseconds(1000) );
165  auto rate = mApp->GetInstantaneousRate();
166  if( rate > 10.0 ) {
167  cout << "Rate: " << rate << "Hz - ready to begin test" <<endl;
168  break;
169  }
170  }
171 
172  // Loop over all thread settings in set
173  map< uint32_t, vector<float> > samples;
174  map< uint32_t, std::pair<float,float> > rates; // key=nthreads val.first=rate in Hz, val.second=rms of rate in Hz
175  for (uint32_t nthreads=mMinThreads; nthreads<=mMaxThreads; nthreads+=mThreadStep){
176  cout << "Setting NTHREADS = " << nthreads << " ..." <<endl;
177  tm->SetNJThreads( nthreads );
178 
179  // Loop for at most 60 seconds waiting for the number of threads to update
180  for(int i=0; i<60; i++){
181  std::this_thread::sleep_for( std::chrono::milliseconds(1000) );
182  if( tm->GetNJThreads() == nthreads ) break;
183  }
184 
185  // Acquire mNsamples instantaneous rate measurements. The
186  // GetInstantaneousRate method will only update every 0.5
187  // seconds so we just wait for 1 second between samples to
188  // ensure independent measurements.
189  double sum = 0;
190  double sum2 = 0;
191  for(uint32_t isample=0; isample<mNsamples; isample++){
192  std::this_thread::sleep_for( std::chrono::milliseconds(1000) );
193  auto rate = mApp->GetInstantaneousRate();
194  samples[nthreads].push_back(rate);
195 
196  sum += rate;
197  sum2 += rate*rate;
198  double N = (double)(isample+1);
199  double avg = sum/N;
200  double rms = sqrt( (sum2 + N*avg*avg - 2.0*avg*sum)/N );
201  rates[nthreads].first = avg; // overwrite with updated value after each sample
202  rates[nthreads].second = rms; // overwrite with updated value after each sample
203 
204  cout << "nthreads=" << tm->GetNJThreads() << " rate=" << rate << "Hz";
205  if( N>1 ) cout << " (avg = " << avg << " +/- " << rms/sqrt(N) << " Hz)";
206  cout << endl;
207  }
208  }
209 
210  // Write results to files
211  LOG_INFO(mLogger) << "Writing test results to: " << mOutputDirName << LOG_END;
212  mkdir(mOutputDirName.c_str(), S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH);
213 
214  std::ofstream ofs1(mOutputDirName+"/samples.dat");
215  ofs1 << "# nthreads rate" << endl;
216  for( auto p : samples ){
217  auto nthreads = p.first;
218  for( auto rate: p.second ) ofs1 << std::setw(7) << nthreads << " " << std::setw(12) << std::setprecision(1) << std::fixed << rate << endl;
219  }
220  ofs1.close();
221 
222  std::ofstream ofs2(mOutputDirName+"/rates.dat");
223  ofs2 << "# nthreads avg_rate rms" << endl;
224  for( auto p : rates ){
225  auto nthreads = p.first;
226  auto avg_rate = p.second.first;
227  auto rms = p.second.second;
228  ofs2 << std::setw(7 ) << nthreads << " ";
229  ofs2 << std::setw(12) << std::setprecision(1) << std::fixed << avg_rate << " ";
230  ofs2 << std::setw(10) << std::setprecision(1) << std::fixed << rms << endl;
231  }
232  ofs2.close();
233 
234  CopyToOutputDir("${JANA_HOME}/src/plugins/JTest/plot_rate_vs_nthreads.py");
235 
236  cout << "Testing finished" << endl;
237  mApp->Quit();
238 }

241 // CopyToOutputDir
242 //
243 // Copy the specified output file to the currently set mOutputDirName.
244 // Strings with a format of ${Name} will be substituted with the
245 // corresponding environment variable.
246 //---------------------------------
247 void JTestMain::CopyToOutputDir(std::string filename)
248 {
249  // Substitute environment variables in given filename
250  string new_fname = filename;
251  while( auto pos_start=new_fname.find("${") != new_fname.npos ){
252  auto pos_end = new_fname.find( "}", pos_start+3 );
253  if( pos_end != new_fname.npos ){
254 
255  string envar_name = new_fname.substr(pos_start+1, pos_end-pos_start-1);
256  LOG_DEBUG(mLogger) << "Looking for env var '" << envar_name
257  << "'" << LOG_END;
258 
259  auto envar = getenv( envar_name.c_str() );
260  if( envar ) {
261  new_fname.replace( pos_start-1, pos_end+2-pos_start, envar);
262  }else{
263  LOG_ERROR(mLogger) << "Environment variable '"
264  << envar_name
265  << "' not set. Cannot copy "
266  << filename << LOG_END;
267  return;
268  }
269  }else{
270  LOG_ERROR(mLogger) << "Error in string format: "
271  << filename << LOG_END;
272  }
273  }
274 
275  // Extract filename without path
276  string base_fname = new_fname;
277  if( auto pos = base_fname.rfind("/") ) base_fname.erase(0, pos);
278 
279  // Copy file
280  LOG_INFO(mLogger) << "Copying " << new_fname << " -> " << mOutputDirName << LOG_END;
281  std::ifstream src(new_fname, std::ios::binary);
282  std::ofstream dst(mOutputDirName + "/" + base_fname, std::ios::binary);
283  dst << src.rdbuf();
284 }
