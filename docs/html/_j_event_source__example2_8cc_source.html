61 //-------------------------------------------------------------------------
62 // GetObjects
63 //-------------------------------------------------------------------------
64 bool JEventSource_example2::GetObjects(const std::shared_ptr<const JEvent>& aEvent, JFactory* aFactory)
65 {
66  // This will get called for every type of object requested for the event
67  // to see if the objects are available from the source. In this example,
68  // the source only provides one type of object: MyHit. If this is not the
69  // type being requested, then return "false" immediately.
70  if( aFactory->GetName() != "MyHit" ) return false;
71 
72  // Here, one would normally get a buffer stored in the JEvent and parse it
73  // to make the low-level objects (e.g. MyHit, ...). For this example though,
74  // we just generate some hit objects using random numbers.
75  static thread_local std::default_random_engine generator;
76 
77  // Determine how many MyHit objects this hit will have
78  std::normal_distribution<double> distribution(50.0,10.0);
79  size_t Nhits = 0;
80  while(Nhits<10) Nhits = distribution(generator);
81 
82  // Create MyHit objects
83  vector<MyHit*> hits;
84  for(size_t i=0; i<Nhits; i++){
85  double x = distribution(generator)/10.0;
86  double E = distribution(generator)*2.0;
87  double t = distribution(generator)/3.0;
88  hits.push_back( new MyHit( x, E, t) );
89  }
90 
91  // Copy pointers into factory, transferring ownership. In this case, the
92  // factory is just used as a container and has no algorithm. See the
93  // MyCluster class for an example of a factory with an algorithm
94  aFactory->Set( hits );
95 
96  // Return true to indicate that this is a type of object the source can provide.
97  return true;
98 }
