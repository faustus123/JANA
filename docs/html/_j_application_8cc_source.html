180 void JApplication::AttachPlugins(void)
181 {
182  /// Loop over list of plugin names added via AddPlugin() and
183  /// actually attach and intiailize them. See AddPlugin method
184  /// for more.
185 
186  bool printPaths = false;
187  GetJParameterManager()->SetDefaultParameter(
188  "JANA:DEBUG_PLUGIN_LOADING",
189  printPaths,
190  "Trace the plugin search path and display any loading errors");
191 
192  // In order to give priority to factories added via plugins,
193  // the list of factory generators needs to be cleared so
194  // those added from plugins will be at the front of the list.
195  // We make a copy of the existing generators first so we can
196  // append them back to the end of the list before exiting.
197  // Similarly for event source generators and calibration generators.
198  vector<JEventSourceGenerator*> my_eventSourceGenerators;
199  _eventSourceManager->GetJEventSourceGenerators(my_eventSourceGenerators);
200  vector<JFactoryGenerator*> my_factoryGenerators = _factoryGenerators;
201  vector<JCalibrationGenerator*> my_calibrationGenerators = _calibrationGenerators;
202  _eventSourceManager->ClearJEventSourceGenerators();
203  _factoryGenerators.clear();
204  _calibrationGenerators.clear();
205 
206  // The JANA_PLUGIN_PATH specifies directories to search
207  // for plugins that were explicitly added through AddPlugin(...).
208  // Multiple directories can be specified using a colon(:) separator.
209  const char *jpp = getenv("JANA_PLUGIN_PATH");
210  if(jpp){
211  stringstream ss(jpp);
212  string path;
213  while(getline(ss, path, ':')) AddPluginPath( path );
214  }
215 
216  // Default plugin search path
217  AddPluginPath(".");
218  if(const char *ptr = getenv("JANA_HOME")) AddPluginPath(string(ptr) + "/plugins");
219 
220  // Add plugins specified via PLUGINS configuration parameter
221  // (comma separated list).
222  try{
223  string plugins;
224  GetJParameterManager()->GetParameter("PLUGINS", plugins);
225  stringstream ss(plugins);
226  string p;
227  while(getline(ss, p, ',')) _plugins.push_back(p);
228  }catch(...){}
229 
230  // Loop over plugins
231  stringstream err_mess;
232  for(string plugin : _plugins){
233  // Sometimes, the user will include the ".so" suffix in the
234  // plugin name. If they don't, then we add it here.
235  if(plugin.substr(plugin.size()-3)!=".so")plugin = plugin+".so";
236 
237  // Loop over paths
238  bool found_plugin=false;
239  for(string path : _plugin_paths){
240  string fullpath = path + "/" + plugin;
241  LOG_TRACE(_logger, printPaths) << "Looking for '" << fullpath << "' ...." << LOG_END;
242  if (access(fullpath.c_str(), F_OK) != -1) {
243  LOG_TRACE(_logger, printPaths) << "Found!" << LOG_END;
244  try{
245  AttachPlugin(fullpath.c_str(), printPaths);
246  found_plugin=true;
247  break;
248  } catch(...) {
249  err_mess << "Tried to attach: \"" << fullpath << "\"" << endl;
250  err_mess << " -- error message: " << dlerror() << endl;
251  continue;
252  }
253  }
254  LOG_TRACE(_logger, printPaths) << "Failed to attach '" << fullpath << "'" << LOG_END;
255  }
256 
257  // If we didn't find the plugin, then complain and quit
258  if(!found_plugin){
259 
260  LOG_FATAL(_logger) << "\n*** Couldn't find plugin '" << plugin << "'! ***\n" <<
261  "*** Make sure the JANA_PLUGIN_PATH environment variable is set correctly.\n" <<
262  "*** To see paths checked, set JANA:DEBUG_PLUGIN_LOADING=1\n"<<
263  "*** Some hints to the error follow:\n\n" << err_mess.str() << LOG_END;
264 
265  exit(-1);
266  }
267  }
268 
269  // Append generators back onto appropriate lists
270  for(auto sGenerator : my_eventSourceGenerators)
271  _eventSourceManager->AddJEventSourceGenerator(sGenerator);
272  _factoryGenerators.insert(_factoryGenerators.end(), my_factoryGenerators.begin(), my_factoryGenerators.end());
273  _calibrationGenerators.insert(_calibrationGenerators.end(), my_calibrationGenerators.begin(), my_calibrationGenerators.end());
274 }

278 //---------------------------------
279 void JApplication::AttachPlugin(string soname, bool verbose)
280 {
281  /// Attach a plugin by opening the shared object file and running the
282  /// InitPlugin_t(JApplication* app) global C-style routine in it.
283  /// An exception will be thrown if the plugin is not successfully opened.
284  /// Users will not need to call this directly since it is called automatically
285  /// from Initialize().
286  ///
287  /// @param soname name of shared object file to attach. This may include
288  /// an absolute or relative path.
289  ///
290  /// @param verbose if set to true, failed attempts will be recorded via the
291  /// JLog. Default is false so JANA can silently ignore files
292  /// that are not valid plugins.
293  ///
294 
295  // Open shared object
296  void* handle = dlopen(soname.c_str(), RTLD_LAZY | RTLD_GLOBAL | RTLD_NODELETE);
297  if(!handle){
298  LOG_TRACE(_logger, verbose) << dlerror() << LOG_END;
299  throw "dlopen failed";
300  }
301 
302  // Look for an InitPlugin symbol
303  typedef void InitPlugin_t(JApplication* app);
304  InitPlugin_t *plugin = (InitPlugin_t*)dlsym(handle, "InitPlugin");
305  if(plugin){
306  LOG_INFO(_logger) << "Initializing plugin \"" << soname << "\"" << LOG_END;
307  (*plugin)(this);
308  _sohandles.push_back(handle);
309  }else{
310  dlclose(handle);
311  LOG_TRACE(_logger, verbose) << "Nothing useful found in " << soname << LOG_END;
312  }
313 }

318 void JApplication::AddPlugin(string plugin_name)
319 {
320  /// Add the specified plugin to the list of plugins to be
321  /// attached. This only records the name. The plugin is not
322  /// actually attached until AttachPlugins() is called (typically
323  /// from Initialize() which is called from Run()).
324  /// This will check if the plugin already exists in the list
325  /// of plugins to attach and will not add it a second time
326  /// if it is already there. This may be important if the order
327  /// of plugins is important. It is left to the user to handle
328  /// in those cases.
329  ///
330  /// @param plugin_name name of the plugin. Do not include the
331  /// ".so" or ".dylib" suffix in the name.
332  /// The path to the plugin will be searched
333  /// from the JANA_PLUGIN_PATH envar.
334  ///
335  for( string &n : _plugins) if( n == plugin_name ) return;
336  _plugins.push_back(plugin_name);
337 }

342 void JApplication::AddPluginPath(string path)
343 {
344  /// Add a path to the directories searched for plugins. This
345  /// should not include the plugin name itself. This only has
346  /// an effect when called before AttachPlugins is called
347  /// (i.e. before Run is called).
348  /// n.b. if this is called with a path already in the list,
349  /// then the call is silently ignored.
350  ///
351  /// Generally, users will set the path via the JANA_PLUGIN_PATH
352  /// environment variable and won't need to call this method. This
353  /// may be called if it needs to be done programmatically.
354  ///
355  /// @param path directory to search fpr plugins.
356  ///
357  for( string &n : _plugin_paths) if( n == path ) return;
358  _plugin_paths.push_back(path);

377 void JApplication::Initialize(void)
378 {
379  /// Initialize the application in preparation for data processing.
380  /// This is called by the Run method so users will usually not
381  /// need to call this directly.
382 
383  // Attach all plugins
384  AttachPlugins();
385 
386  // Create all event sources
387  _eventSourceManager->CreateSources();
388 
389  // Get factory generators from event sources
390  std::deque<JEventSource*> sEventSources;
391  _eventSourceManager->GetUnopenedJEventSources(sEventSources);
392  std::unordered_set<std::type_index> sSourceTypes;
393  for(auto sSource : sEventSources)
394  {
395  auto sTypeIndex = sSource->GetDerivedType();
396  if(sSourceTypes.find(sTypeIndex) != std::end(sSourceTypes))
397  continue; //same type as before: duplicate factories!
398 
399  auto sGenerator = sSource->GetFactoryGenerator();
400  if(sGenerator != nullptr)
401  _factoryGenerators.push_back(sGenerator);
402  }
403 
404  //Prepare for running: Open event sources and prepare task queues for them
405  _eventSourceManager->OpenInitSources();
406  _threadManager->PrepareQueues();
407 }

412 void JApplication::PrintFinalReport(void)
413 {
414  //Get queues
415  std::vector<JThreadManager::JEventSourceInfo*> sRetiredQueues;
416  _threadManager->GetRetiredSourceInfos(sRetiredQueues);
417  std::vector<JThreadManager::JEventSourceInfo*> sActiveQueues;
418  _threadManager->GetActiveSourceInfos(sActiveQueues);
419  auto sAllQueues = sRetiredQueues;
420  sAllQueues.insert(sAllQueues.end(), sActiveQueues.begin(), sActiveQueues.end());
421 
422 
423  // Get longest JQueue name
424  uint32_t sSourceMaxNameLength = 0, sQueueMaxNameLength = 0;
425  for(auto& sSourceInfo : sAllQueues)
426  {
427  auto sSource = sSourceInfo->mEventSource;
428  auto sSourceLength = sSource->GetName().size();
429  if(sSourceLength > sSourceMaxNameLength)
430  sSourceMaxNameLength = sSourceLength;
431 
432  std::map<JQueueSet::JQueueType, std::vector<JQueue*>> sSourceQueues;
433  sSourceInfo->mQueueSet->GetQueues(sSourceQueues);
434  for(auto& sTypePair : sSourceQueues)
435  {
436  for(auto sQueue : sTypePair.second)
437  {
438  auto sLength = sQueue->GetName().size();
439  if(sLength > sQueueMaxNameLength)
440  sQueueMaxNameLength = sLength;
441  }
442  }
443  }
444  sSourceMaxNameLength += 2;
445  if(sSourceMaxNameLength < 8) sSourceMaxNameLength = 8;
446  sQueueMaxNameLength += 2;
447  if(sQueueMaxNameLength < 7) sQueueMaxNameLength = 7;
448 
449  jout << std::endl;
450  jout << "Final Report" << std::endl;
451  jout << std::string(sSourceMaxNameLength + 12 + sQueueMaxNameLength + 9, '-') << std::endl;
452  jout << "Source" << std::string(sSourceMaxNameLength - 6, ' ') << " Nevents " << "Queue" << std::string(sQueueMaxNameLength - 5, ' ') << "NTasks" << std::endl;
453  jout << std::string(sSourceMaxNameLength + 12 + sQueueMaxNameLength + 9, '-') << std::endl;
454  std::size_t sSrcIdx = 0;
455  for(auto& sSourceInfo : sAllQueues)
456  {
457  // Flag to prevent source name and number of events from
458  // printing more than once.
459  bool sSourceNamePrinted = false;
460 
461  // Place "*" next to names of active sources
462  string sFlag;
463  if( sSrcIdx++ >= sRetiredQueues.size() ) sFlag="*";
464 
465  auto sSource = sSourceInfo->mEventSource;
466  std::map<JQueueSet::JQueueType, std::vector<JQueue*>> sSourceQueues;
467  sSourceInfo->mQueueSet->GetQueues(sSourceQueues);
468  for(auto& sTypePair : sSourceQueues)
469  {
470  for(auto sQueue : sTypePair.second)
471  {
472  string sSourceName = sSource->GetName()+sFlag;
473 
474  if( sSourceNamePrinted ){
475  jout << string(sSourceMaxNameLength + 12, ' ');
476  }else{
477  sSourceNamePrinted = true;
478  jout << sSourceName << string(sSourceMaxNameLength - sSourceName.size(), ' ')
479  << std::setw(10) << sSource->GetNumEventsProcessed() << " ";
480  }
481 
482  jout << sQueue->GetName() << string(sQueueMaxNameLength - sQueue->GetName().size(), ' ')
483  << sQueue->GetNumTasksProcessed() << std::endl;
484  }
485  }
486  }
487  if( !sActiveQueues.empty() ){
488  jout << std::endl;
489  jout << "(*) indicates sources that were still active" << std::endl;
490  }
491 
492  jout << std::endl;
493  jout << "Total events processed: " << GetNeventsProcessed() << " (~ " << Val2StringWithPrefix( GetNeventsProcessed() ) << "evt)" << std::endl;
494  jout << "Integrated Rate: " << Val2StringWithPrefix( GetIntegratedRate() ) << "Hz" << std::endl;
495  jout << std::endl;
496 
497  // Optionally print more info if user requested it:
498  bool print_extended_report = false;
499  _pmanager->SetDefaultParameter("JANA:EXTENDED_REPORT", print_extended_report);
500  if( print_extended_report ){
501  jout << std::endl;
502  jout << "Extended Report" << std::endl;
503  jout << std::string(sSourceMaxNameLength + 12 + sQueueMaxNameLength + 9, '-') << std::endl;
504  jout << " Num. plugins: " << _plugins.size() <<std::endl;
505  jout << " Num. plugin paths: " << _plugin_paths.size() <<std::endl;
506  jout << " Num. factory generators: " << _factoryGenerators.size() <<std::endl;
507  jout << "Num. calibration generators: " << _calibrationGenerators.size() <<std::endl;
508  jout << " Num. event processors: " << mNumProcessorsAdded <<std::endl;
509  jout << " Num. factory sets: " << mFactorySetPool.Get_PoolSize() << " (max. " << mFactorySetPool.Get_MaxPoolSize() << ")" << std::endl;
510  jout << " Num. config. params.: " << _pmanager->GetNumParameters() <<std::endl;
511  jout << " Num. threads: " << _threadManager->GetNJThreads() <<std::endl;
512  }
513 }

539 // Run
540 //---------------------------------
541 void JApplication::Run(uint32_t nthreads)
542 {
543  // Set number of threads
544  try{
545  string snthreads = GetParameterValue<string>("NTHREADS");
546  if( snthreads == "Ncores" ){
547  nthreads = _threadManager->GetNcores();
548  }else{
549  stringstream ss(snthreads);
550  ss >> nthreads;
551  }
552  }catch(...){}
553  if( nthreads==0 ) nthreads=1;
554 
555  // Setup all queues and attach plugins
556  Initialize();
557 
558  // If something went wrong in Initialize() then we may be quitting early.
559  if(_quitting) return;
560 
561  // Create all remaining threads (one may have been created in Init)
562  jout << "Creating " << nthreads << " processing threads ..." << endl;
563  _threadManager->CreateThreads(nthreads);
564 
565  // Optionally set thread affinity
566  try{
567  int affinity_algorithm = 0;
568  _pmanager->SetDefaultParameter("AFFINITY", affinity_algorithm, "Set the thread affinity algorithm. 0=none, 1=sequential, 2=core fill");
569  _threadManager->SetThreadAffinity( affinity_algorithm );
570  }catch(...){}
571 
572  // Print summary of all config parameters (if any aren't default)
573  GetJParameterManager()->PrintParameters(true);
574 
575  // Start all threads running
576  jout << "Start processing ..." << endl;
577  mRunStartTime = std::chrono::high_resolution_clock::now();
578  _threadManager->RunThreads();
579 
580  // Monitor status of all threads
581  while( !_quitting ){
582 
583  // If we are finishing up (all input sources are closed, and are waiting for all events to finish processing)
584  // This flag is used by the integrated rate calculator
585  // The JThreadManager is in charge of telling all the threads to end
586  if(!_draining_queues)
587  _draining_queues = _eventSourceManager->AreAllFilesClosed();
588 
589  // Check if all threads have finished
590  if(_threadManager->AreAllThreadsEnded())
591  {
592  std::cout << "All threads have ended.\n";
593  break;
594  }
595 
596  // Sleep a few cycles
597  std::this_thread::sleep_for( std::chrono::milliseconds(500) );
598 
599  // Print status
600  if( _ticker_on ) PrintStatus();
601  }
602 
603  // Join all threads
604  jout << "Event processing ended. " << endl;
605  if (!_skip_join) {
606  cout << "Merging threads ..." << endl;
607  _threadManager->JoinThreads();
608  }
609 
610  // Delete event processors
611  for(auto sProcessor : _eventProcessors){
612  sProcessor->Finish(); // (this may not be necessary since it is always next to destructor)
613  delete sProcessor;
614  }
615  _eventProcessors.clear();
616 
617  // Report Final numbers
618  PrintFinalReport();
619 }

760 //---------------------------------
761 // GetCPU
762 //---------------------------------
763 uint32_t JApplication::GetCPU(void)
764 {
765  /// Returns the current CPU the calling thread is running on.
766  /// Note that unless the thread affinity has been set, this may
767  /// change, even before returning from this call. The thread
768  /// affinity of all threads may be fixed by setting the AFFINITY
769  /// configuration parameter at program start up.
770 
771  int cpuid;
772 
773 #ifdef __APPLE__
774 
775  //--------- Mac OS X ---------
776 
777 // From https://stackoverflow.com/questions/33745364/sched-getcpu-equivalent-for-os-x
778 #define CPUID(INFO, LEAF, SUBLEAF) __cpuid_count(LEAF, SUBLEAF, INFO[0], INFO[1], INFO[2], INFO[3])
779 #define GETCPU(CPU) { \
780  uint32_t CPUInfo[4]; \
781  CPUID(CPUInfo, 1, 0); \
782  /* CPUInfo[1] is EBX, bits 24-31 are APIC ID */ \
783  if ( (CPUInfo[3] & (1 << 9)) == 0) { \
784  CPU = -1; /* no APIC on chip */ \
785  } \
786  else { \
787  CPU = (unsigned)CPUInfo[1] >> 24; \
788  } \
789  if (CPU < 0) CPU = 0; \
790  }
791 
792  GETCPU(cpuid);
793 #else // __APPLE__
794 
795  //--------- Linux ---------
796  cpuid = sched_getcpu();
797 
798 #endif // __APPLE__
799 
800 
801  return cpuid;
802 }

829 // UpdateResourceLimits
830 //---------------------------------
831 void JApplication::UpdateResourceLimits(void)
832 {
833  /// Used internally by JANA to adjust the maximum size of resource
834  /// pools after changing the number of threads.
835 
836  // OK, this is tricky. The max size of the JFactorySet resource pool should
837  // be at least as big as how many threads we have. Factory sets may also be
838  // attached to JEvent objects in queues that are not currently being acted
839  // on by a thread so we actually need the maximum to be larger if we wish to
840  // prevent constant allocation/deallocation of factory sets. If the factory
841  // sets are large, this will cost more memory, but should save on CPU from
842  // allocating less often and not having to call the constructors repeatedly.
843  // The exact maximum is hard to determine here. We set it to twice the number
844  // of threads which should be sufficient. The user should be given control to
845  // adjust this themselves in the future, but or now, this should be OK.
846  auto nthreads = _threadManager->GetNJThreads();
847  mFactorySetPool.Set_ControlParams( nthreads*2, 10 );
848 }

870 float JApplication::GetIntegratedRate(void)
871 {
872  /// Returns the total integrated rate so far in Hz since
873  /// Run was called.
874 
875  static float last_R = 0.0;
876 
877  auto now = std::chrono::high_resolution_clock::now();
878  std::chrono::duration<float> delta_t = now - mRunStartTime;
879  float delta_t_seconds = delta_t.count();
880  float delta_N = (float)GetNeventsProcessed();
881

883  last_R = delta_N / delta_t_seconds;
884  }
885

890 // GetInstantaneousRate
891 //---------------------------------
892 float JApplication::GetInstantaneousRate(void)
893 {
894  auto now = std::chrono::high_resolution_clock::now();
895  uint64_t N = GetNeventsProcessed();
896  static auto last_t = now;
897  static uint64_t last_N = N;
898 
899  std::chrono::duration<float> delta_t = now - last_t;
900  float delta_t_seconds = delta_t.count();
901  float delta_N = (float)(GetNeventsProcessed() - last_N);
902 
903  static float last_R = 0.0;
904  if( delta_t_seconds >= 0.5) {
905  last_R = delta_N / delta_t_seconds;
906  last_t = now;
907  last_N = N;
908  }
909 
910  return last_R;
911 }

949 string JApplication::Val2StringWithPrefix(float val)
950 {
951  /// Return the value as a string with the appropriate latin unit prefix
952  /// appended.
953  /// Values returned are: "G", "M", "k", "", "u", and "m" for
954  /// values of "val" that are: >1.5E9, >1.5E6, >1.5E3, <1.0E-7, <1.0E-4, 1.0E-1
955  /// respectively.
956  const char *units = "";
957  if(val>1.5E9){
958  val/=1.0E9;
959  units = "G";
960  }else if(val>1.5E6){
961  val/=1.0E6;
962  units = "M";
963  }else if(val>1.5E3){
964  val/=1.0E3;
965  units = "k";
966  }else if(val<1.0E-7){
967  units = "";
968  }else if(val<1.0E-4){
969  val/=1.0E6;
970  units = "u";
971  }else if(val<1.0E-1){
972  val/=1.0E3;
973  units = "m";
974  }
975 
976  char str[256];
977  sprintf(str,"%3.1f %s", val, units);
978 
979  return string(str);
980 }
