52 extern "C"{
53 void InitPlugin(JApplication *app){
54  InitJANAPlugin(app);
55 
56  // Launch thread to handle Python interpreter.
57  std::thread thr(JANA_PythonModuleInit, app);
58  thr.detach();
59 
60  // Wait for interpreter to initialize and possibly run script.
61  // This allows more control from python by stalling the initialization
62  // so it has a chance to modify things a bit before initialization
63  // completes and data processing starts.
64  while( !PY_INITIALIZED ) std::this_thread::sleep_for (std::chrono::milliseconds(100));
65 }
66 } // "C"

69 // The following effectively make a template out of "PV" so it can
70 // be used to convert all types to PyObjects
71 template<typename T> PyObject* PV(T v){ return Py_BuildValue("i", (int)v); }
72 template<> PyObject* PV<int>(int v){ return Py_BuildValue("i", v); }
73 template<> PyObject* PV<long>(long v){ return Py_BuildValue("l", v); }
74 template<> PyObject* PV<float>(float v){ return Py_BuildValue("f", v); }
75 template<> PyObject* PV<double>(double v){ return Py_BuildValue("d", v); }
76 template<> PyObject* PV<string>(string v){ return Py_BuildValue("s", v.c_str()); }
77 template<> PyObject* PV<const char*>(const char* v){ return Py_BuildValue("s", v); }

79 // 1-D container types (vector, list, set, ...)
80 template<typename T>
81 PyObject* PVlist( T& c ){
82  PyObject *PList = PyList_New( c.size() );
83  std::size_t i=0;
84  for( auto v : c ) PyList_SET_ITEM( PList, i++, PV(v));
85  return PList;
86 }

368 // JANA_PythonModuleInit

370 void JANA_PythonModuleInit(JApplication *sApp)
371 {
372  /// This will initialize the embedded Python interpreter and import the
373  /// JANA module (defined by the routines in this file). This gets called
374  /// when the janapy plugin gets initialized. It will then automatically
375  /// look for a file whose name is given by either the JANA_PYTHON_FILE
376  /// config. parameter or environment variable in that order. If neither
377  /// exists, it looks for a file name "jana.py" in the current working
378  /// directory and if found, executes that. If you do not wish to execute
379  /// a python file, then you can set the JANA_PYTHON_FILE value to an empty
380  /// string (or better yet, just not attach the janapy plugin!).
381  ///
382  /// Note that janapy creates a dedicated thread that the python interpreter
383  /// runs in.
384  ///
385  /// IMPORTANT: The janapy InitPlugin routine will block (resulting in the whole
386  /// JANA system pausing) until either this routine finishes or the python
387  /// script it invokes calls "jana.Run()". This is to give the python script
388  /// a chance to modify running conditions prior to event processing starting.
389  /// If the python script intends to run throughout the life of the process,
390  /// then it MUST call jana.Run() at some point. If the script is small and only
391  /// runs for a short time, then you don't need to call it since it will be
392  /// called automatically when the script ends.
393 
394  // Check if interpreter is already initialized and do nothing at all if it is.
395  if( Py_IsInitialized() ){
396  jout << "Python already initialized! Skipping initialization of janapy!" << std::endl;
397  PY_INITIALIZED = true;
398  return;
399  }
400 
401  // Initialize interpreter and register the jana module
402  jout << "Initializing embedded Python ... " << std::endl;
403  PyEval_InitThreads();
404  Py_Initialize();
405  Py_InitModule("jana", JANAPYMethods);
406 
407  // Get name of python file to execute
408  string fname = "jana.py";
409  try{
410  fname = sApp->GetParameterValue<string>("JANA_PYTHON_FILE");
411  }catch(...){
412  auto JANA_PYTHON_FILE = getenv("JANA_PYTHON_FILE");
413  if( JANA_PYTHON_FILE ) fname = JANA_PYTHON_FILE;
414  }
415 
416 // PyGILState_STATE gstate = PyGILState_Ensure();
417 // PyEval_ReleaseLock();
418 // PyThreadState *_save = PyEval_SaveThread();
419 // Py_BEGIN_ALLOW_THREADS
420 
421  auto fil = std::fopen(fname.c_str(), "r");
422  if( fil ) {
423  jout << "Executing Python script: " << fname << " ..." << std::endl;
424  const char *argv = fname.c_str();
425  PySys_SetArgv( 1, (char**)&argv );
426  PyRun_AnyFileEx( fil, NULL, 1 );
427  }else if( fname != "jana.py" ){
428  jerr << std::endl << "Unable to open \"" << fname << "\"! Quitting." << std::endl << std::endl;
429  sApp->Quit();
430  PY_INITIALIZED = true;
431  std::this_thread::sleep_for(std::chrono::seconds(1));
432  exit(-1);
433  }
434 
435 // Py_END_ALLOW_THREADS
436 
437  PY_INITIALIZED = true;
438 }
