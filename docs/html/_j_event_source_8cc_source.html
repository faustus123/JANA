85 // GetNumEventsProcessed
86 //---------------------------------
87 std::size_t JEventSource::GetNumEventsProcessed(void) const
88 {
89  /// Return total number of events processed by this source.
90  /// This is a little tricky. A source may consist of multiple
91  /// queues that expand one "event" into many. This returns
92  /// the count returned by the GetNumEventsProcessed() of the
93  /// queue pointed to by the mEventQueue member of this source
94  /// if it is set. This presumably is the last queue in the
95  /// chain and what the caller is most interested in. If that
96  /// is not set, then the value of the member mEventsProcessed
97  /// is returned. This is a count of the number of events
98  /// processed through all queues, including any internal ones.
99  /// In those situations, this will likely be double counting.
100  ///
101  /// Note that in the first case, the number is how many events
102  /// were removed from the last queue, but some of those may
103  /// still be processing. Thus, it might overcount how many were
104  /// actually completed. This is only a problem if this is called
105  /// while event processing is still ongoing.
106  if( mEventQueue != nullptr ) return mEventQueue->GetNumTasksProcessed();
107 
108  return mEventsProcessed;
109 }

120 // GetProcessEventTasks
121 //---------------------------------
122 std::vector<std::shared_ptr<JTaskBase> > JEventSource::GetProcessEventTasks(std::size_t aNumTasks)
123 {
124  /// This version is called by JThread.
125  /// This will attempt to read aNumTasks from the source, wrapping each in a JTask
126  /// so it can be added to this source's Event queue.
127 
128  //If file closed, return dummy pair
129  if(mExhausted) return std::vector<std::shared_ptr<JTaskBase>>();
130 
131  // Optionally limit number of events read from this source
132  if( (mMaxEventsToRead!=0) && ((mEventsRead + aNumTasks)>mMaxEventsToRead) ){
133  aNumTasks = mMaxEventsToRead - mEventsRead;
134  }
135 
136  //Initialize things before locking
137  std::vector<std::shared_ptr<const JEvent>> sEvents;
138  sEvents.reserve(aNumTasks);
139 
140  //Attempt to acquire atomic lock
141  bool sExpected = false;
142  if(!mGettingEvent.compare_exchange_strong(sExpected, true)){
143  // failed to get lock. Return empty container
144  return std::vector<std::shared_ptr<JTaskBase> >();
145  }
146 
147  //Lock aquired, get the events
148  for(std::size_t si = 0; si < aNumTasks; si++)
149  {
150  // Read an event from the source. Anything other than a successful read
151  // throws an exception (we never need to check for kSUCCESS)
152  try{
153  std::shared_ptr<const JEvent> jevent = GetEvent();
154  std::const_pointer_cast<JEvent>(jevent)->SetJEventSource(this); // don't tell the C++ police!
155  std::const_pointer_cast<JEvent>(jevent)->SetJApplication(mApplication); // don't tell the C++ police!
156  sEvents.push_back(std::move(jevent));
157  }catch(RETURN_STATUS ret_status){
158  switch(ret_status){
159  case RETURN_STATUS::kNO_MORE_EVENTS:
160  case RETURN_STATUS::kERROR:
161  mExhausted = true;
162  break;
163  case RETURN_STATUS::kBUSY:
164  case RETURN_STATUS::kTRY_AGAIN:
165  default:
166  break;
167  }
168  break; // exception caught so don't try reading any more events right now
169  }catch(...){
170  mExhausted = true;
171  break; // un-expected exception caught
172  }
173  }
174 
175  //Done with the lock: Unlock
176  mGettingEvent = false;
177 
178  //Make tasks for analyzing the events
179  std::vector<std::shared_ptr<JTaskBase> > sTasks;
180  mEventsRead.fetch_add(sEvents.size());
181  for(auto& sEvent : sEvents) sTasks.push_back(GetProcessEventTask(std::move(sEvent)));
182  mTasksCreated.fetch_add(sTasks.size());
183 
184 
185  //Return the tasks
186  return sTasks;
187 }
